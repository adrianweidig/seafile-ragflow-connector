# =============================================================================
# ADMIN- UND ENTWICKLERDOKUMENTATION
# =============================================================================
# Zweck dieser Pipe
# -----------------
# Diese OpenWebUI-Pipe verbindet OpenWebUI mit einem eigenen Connector-Proxy,
# der wiederum RAGFlow beziehungsweise ein RAGFlow-Dataset/Chat anspricht.
#
# Wichtig: Die Pipe ist als Audit-RAG-Pipe gebaut. Retrieval-Treffer sind
# nicht nur Anhang, sondern die Beweiskette der Antwort. Die finale
# Chat-Nachricht soll zuerst eine echte KI-Antwort sein und jede wesentliche
# Aussage mit Quellenmarken wie [S1] belegen, danach folgt eine sichtbare
# Nachweistabelle plus native Citation-Events.
#
# Typische Datenflusskette:
#
#   OpenWebUI Chat
#        -> diese Pipe
#        -> Connector-Proxy: /api/openwebui/proxy/chat
#        -> RAGFlow Dataset/Chat
#        -> generierte Antwort + Quellen
#        -> OpenWebUI Antwort mit Citations
#
# Wo werden Einstellungen vorgenommen?
# ------------------------------------
# In OpenWebUI öffnest du die Funktion/Pipe und pflegst die Werte im Bereich
# "Valves". Valves sind OpenWebUI-konforme Laufzeitparameter. Sie erlauben, die
# Pipe ohne Codeänderung auf verschiedene Server, Datasets, Modelle und UX-Modi
# anzupassen.
#
# Minimal notwendige Einstellungen:
#   CONNECTOR_PROXY_BASE_URL       URL deines Connector-Proxys
#   CONNECTOR_PROXY_SHARED_SECRET  Bearer-Secret, das auch der Proxy kennt
#   DATASET_ID oder RAGFLOW_CHAT_ID abhängig davon, was dein Proxy erwartet
#   MODEL_ID                       interne Modell-ID in OpenWebUI
#   MODEL_NAME                     schöner Anzeigename im Modellmenü
#
# Empfohlene Chat-RAG-Einstellungen:
#   INJECT_RAG_SYSTEM_PROMPT=true
#   SEND_GENERATIVE_RAG_HINTS=true
#   RETRIEVAL_ONLY_FALLBACK=brief
#   EMIT_CITATION_EVENTS=true
#   APPEND_SOURCE_OVERVIEW=true
#   SOURCE_MARKDOWN_MODE=audit
#   SHOW_SOURCE_SCORES=false
#   SHOW_SOURCE_DEBUG=false
#
# Wann wird ENABLE_ANSWER_SYNTHESIS_FALLBACK benötigt?
# ----------------------------------------------------
# Nur dann, wenn dein Connector-Proxy aktuell ausschließlich Retrieval-Treffer
# zurückliefert, aber keinen generierten Antworttext. Dann kann diese Pipe aus
# den gefundenen Quellen über ein separates OpenAI-kompatibles LLM eine Antwort
# erzeugen. Das ist ein Fallback, nicht der bevorzugte Hauptpfad.
#
# Bevorzugter Hauptpfad:
#   Proxy/RAGFlow liefert selbst ein Feld wie answer, data.answer oder
#   choices[0].message.content.
#
# Sicherheitsgrundsätze
# ---------------------
# - Secrets niemals in Logs, Fehlermeldungen oder Chat-Antworten ausgeben.
# - CONNECTOR_PROXY_VERIFY_SSL nur für lokale Tests deaktivieren.
# - SHOW_SOURCE_DEBUG nur für Admin-Debugging aktivieren, weil interne IDs und
#   technische Metadaten sichtbar werden können.
# - ERROR_VERBOSITY=debug nur temporär nutzen.
#
# UX-Grundsätze für die Chat-Ausgabe
# ----------------------------------
# - Antwort zuerst, Quellen danach.
# - Keine Roh-Chunks als Ersatz für eine Antwort.
# - Native Citation-Events für klickbare Quellen verwenden.
# - Markdown-Nachweistabelle im Audit-Modus sichtbar halten.
# - Wenn keine generierte Antwort vorliegt, klar diagnostizieren, warum.
# =============================================================================

class PipeError(Exception):
    """Exception with a safe end-user message and a status-line variant."""

    def __init__(self, user_message: str, *, status: Optional[str] = None) -> None:
        super().__init__(user_message)
        self.user_message = user_message
        self.status = status or user_message

class Pipe:
    class Valves(BaseModel):
        """OpenWebUI-Valves: zentrale Laufzeitkonfiguration der Pipe.

        Jede Einstellung ist absichtlich als Valve modelliert, damit Admins sie
        in OpenWebUI ändern können, ohne das Skript neu zu bearbeiten.

        Konfigurationslogik:
        - Proxy-/RAGFlow-Valves bestimmen, wohin die Anfrage geht.
        - Chat-RAG-Valves bestimmen, ob generative Antworten erzwungen werden.
        - Fallback-Valves bestimmen, was passiert, wenn nur Quellen ankommen.
        - UX-Valves bestimmen, wie Status, Citations und Quellen aussehen.
        - Debug-Valves sind für Fehlersuche gedacht und sollten im Normalbetrieb
          deaktiviert bleiben.
        """

        # ------------------------------------------------------------------
        # 1) Connector- und RAGFlow-Zielsystem
        # ------------------------------------------------------------------
        # ARTIFACT_ID ist optional. Nutze diesen Wert, wenn dein Proxy mehrere
        # Artefakte/Installationen unterscheiden muss. Wenn dein Proxy damit
        # nichts macht, leer lassen.
        ARTIFACT_ID: str = Field(
            default="",
            description="Optionale Artefakt-ID für den Connector/Proxy.",
        )

        # Basis-URL des eigenen Connector-Proxys ohne abschließenden API-Pfad.
        # Die Pipe hängt automatisch PROXY_CHAT_PATH an, standardmäßig:
        # /api/openwebui/proxy/chat
        # Beispiel: https://rag-proxy.example.org
        CONNECTOR_PROXY_BASE_URL: str = Field(
            default="",
            description="Basis-URL des Connector-Proxys, z. B. https://proxy.example.org.",
        )

        # Shared Secret für die Verbindung OpenWebUI -> Connector-Proxy.
        # Der Wert wird als HTTP-Header gesendet:
        # Authorization: Bearer <CONNECTOR_PROXY_SHARED_SECRET>
        # Niemals in Support-Tickets, Logs oder Screenshots veröffentlichen.
        CONNECTOR_PROXY_SHARED_SECRET: str = Field(
            default="",
            description="Bearer-Secret für den Connector-Proxy. Niemals in Logs oder Chat ausgeben.",
        )

        # TLS-Zertifikatsprüfung. Im produktiven Betrieb immer True lassen.
        # False ist nur für lokale Entwicklung mit selbstsignierten Zertifikaten
        # vertretbar. Bei False ist die Transportverbindung anfälliger für
        # Man-in-the-Middle-Risiken.
        CONNECTOR_PROXY_VERIFY_SSL: bool = Field(
            default=True,
            description="TLS-Zertifikate des Connector-Proxys prüfen. Nur für lokale Tests deaktivieren.",
        )

        # Optionaler Pfad zu einer CA-Bundle-Datei, wenn dein Proxy ein Zertifikat
        # einer internen/private CA nutzt. Der Pfad muss aus Sicht der
        # OpenWebUI-Runtime erreichbar sein, nicht aus Sicht deines Browsers.
        # Beispiel: /etc/ssl/certs/company-ca.pem
        CONNECTOR_PROXY_CA_BUNDLE: str = Field(
            default="",
            description="Optionaler Pfad zu einem CA-Bundle für interne Zertifizierungsstellen.",
        )

        # Schaltet ausführlichere Exception-Logs für TLS/HTTP ein. Secrets werden
        # weiterhin nicht geloggt. Nur temporär aktivieren, wenn Verbindungsfehler
        # untersucht werden.
        TLS_DEBUG: bool = Field(
            default=False,
            description="Aktiviert ausführlichere TLS-/HTTP-Fehlerlogs ohne Secrets offenzulegen.",
        )

        # Dataset-ID für RAGFlow. Je nach Proxy-Implementierung kann entweder
        # DATASET_ID, RAGFLOW_CHAT_ID oder beides nötig sein. DATASET_ID ist
        # typisch, wenn der Proxy selbst den Chat-/Retrieval-Flow baut.
        DATASET_ID: str = Field(
            default="",
            description="RAGFlow-Dataset-ID, die der Proxy verwenden soll.",
        )

        # Chat-/Assistant-ID für RAGFlow. Wichtig, wenn der Proxy RAGFlows
        # Chat-API nutzt und dort ein konkreter Chat/Assistant konfiguriert ist.
        # Für echte Chat-RAG-Antworten ist dieser Wert häufig entscheidender als
        # eine reine Dataset-ID.
        RAGFLOW_CHAT_ID: str = Field(
            default="",
            description="RAGFlow-Chat-/Assistant-ID, die der Proxy verwenden soll.",
        )

        # ------------------------------------------------------------------
        # 2) Darstellung als Modell in OpenWebUI
        # ------------------------------------------------------------------
        # Interne Modell-ID, die OpenWebUI für diese Pipe verwendet. Kurz,
        # stabil und ohne Leerzeichen wählen, z. B. seafile-ragflow.
        MODEL_ID: str = Field(
            default="",
            description="Modell-ID, unter der diese Pipe in OpenWebUI erscheint.",
        )

        # Sichtbarer Name im Modellmenü. Hier darf ein schöner, sprechender Name
        # stehen, z. B. "Seafile RAGFlow Chat".
        MODEL_NAME: str = Field(
            default="",
            description="Anzeigename im OpenWebUI-Modellmenü.",
        )

        # Anzahl angeforderter Retrieval-Treffer. Höher ist nicht automatisch
        # besser: Zu viele Chunks können Antworten verwässern, Latenz erhöhen und
        # Quellenübersichten überladen. Für Chat-RAG sind 6 bis 10 meist sinnvoll.
        TOP_K: int = Field(
            default=8,
            ge=1,
            le=30,
            description="Maximale Anzahl angeforderter Retrieval-Treffer.",
        )

        # ------------------------------------------------------------------
        # 3) Chat-RAG-Verhalten: Antwort statt Suchergebnisliste
        # ------------------------------------------------------------------
        # Fügt den RAG_SYSTEM_PROMPT als Systemnachricht in den Chatverlauf ein.
        # Eingeschaltet lassen, wenn der Proxy/RAGFlow sonst dazu tendiert, nur
        # Trefferlisten oder Roh-Chunks zurückzugeben.
        INJECT_RAG_SYSTEM_PROMPT: bool = Field(
            default=True,
            description="Ergänzt einen System-Prompt, der RAGFlow explizit auf Chat-Antworten statt Suchergebnisse ausrichtet.",
        )

        # Admin-feinjustierbarer System-Prompt. Hier wird das Antwortverhalten
        # definiert: Antwort zuerst, Quellen unterstützend, keine Halluzinationen,
        # keine Rohmarker. Nur ändern, wenn Tonalität oder Regeln bewusst angepasst
        # werden sollen.
        RAG_SYSTEM_PROMPT: str = Field(
            default=DEFAULT_RAG_SYSTEM_PROMPT,
            description="System-Prompt für generatives RAG-Verhalten.",
        )

        # Sendet zusätzliche maschinenlesbare Hinweise an den Connector-Proxy.
        # Diese Felder sind absichtlich redundant, weil Proxy-/RAGFlow-Versionen
        # unterschiedliche Parameter auswerten können.
        # Wichtig: Dein Proxy muss diese Felder weiterreichen oder interpretieren,
        # sonst bleibt er eventuell im Retrieval-only-Modus.
        SEND_GENERATIVE_RAG_HINTS: bool = Field(
            default=True,
            description="Sendet zusätzliche Proxy-Hinweise wie generate_answer=true und response_mode=chat.",
        )

        # Verhalten, wenn zwar Quellen zurückkommen, aber kein KI-Antworttext.
        # diagnostic: Beste Admin-Hilfe; erklärt, welche Antwortfelder fehlen.
        # brief:      Kürzerer Nutzerhinweis.
        # sources_only: Minimalhinweis; Quellenblock steht im Vordergrund.
        RETRIEVAL_ONLY_FALLBACK: str = Field(
            default="brief",
            description="Verhalten, wenn der Proxy nur Quellen, aber keinen KI-Text liefert: diagnostic, brief oder sources_only.",
        )

        # ------------------------------------------------------------------
        # 4) Optionaler Antwort-Synthese-Fallback
        # ------------------------------------------------------------------
        # Aktivieren, wenn dein Proxy derzeit nur Retrieval kann. Die Pipe ruft
        # dann ein separates OpenAI-kompatibles Chat-Completions-Modell auf und
        # erzeugt aus den Quellen eine finale Antwort.
        # Empfehlung: Aus lassen, sobald dein Proxy/RAGFlow selbst generative
        # Chat-Antworten liefert. Sonst entsteht unnötige Latenz/Komplexität.
        ENABLE_ANSWER_SYNTHESIS_FALLBACK: bool = Field(
            default=False,
            description="Optional: Wenn der Proxy nur Retrieval liefert, wird aus den Quellen über ein OpenAI-kompatibles LLM eine Antwort generiert.",
        )

        # Base-URL des OpenAI-kompatiblen Fallback-LLMs.
        # Beispiele:
        #   https://openwebui.example.org/api
        #   https://llm-gateway.example.org/v1
        # Die Pipe ergänzt /chat/completions automatisch, falls nötig.
        ANSWER_LLM_BASE_URL: str = Field(
            default="",
            description="OpenAI-kompatible Base-URL für den optionalen Antwort-Synthese-Fallback, z. B. https://openwebui.example.org/api.",
        )

        # API-Key für das Fallback-LLM. Nur nötig, wenn
        # ENABLE_ANSWER_SYNTHESIS_FALLBACK=true ist.
        ANSWER_LLM_API_KEY: str = Field(
            default="",
            description="API-Key für den optionalen Antwort-Synthese-Fallback. Niemals in Logs oder Chat ausgeben.",
        )

        # Modellname für das Fallback-LLM, z. B. gpt-4.1-mini, llama3.1:8b oder
        # der Name, den dein OpenWebUI/LLM-Gateway erwartet.
        ANSWER_LLM_MODEL: str = Field(
            default="",
            description="Modell für den optionalen Antwort-Synthese-Fallback.",
        )

        # Maximale Kontextmenge aus Quellen für den Fallback. Höher bedeutet mehr
        # Kontext, aber mehr Tokenkosten und potenziell langsamere Antworten.
        ANSWER_CONTEXT_MAX_CHARS: int = Field(
            default=12000,
            ge=1000,
            le=60000,
            description="Maximale Kontextlänge aus Quellen für den optionalen Antwort-Synthese-Fallback.",
        )

        # Temperatur für den Fallback. Für RAG niedrig halten, damit Antworten
        # faktenorientiert und stabil bleiben.
        ANSWER_SYNTHESIS_TEMPERATURE: float = Field(
            default=0.2,
            ge=0.0,
            le=1.0,
            description="Temperatur für den optionalen Antwort-Synthese-Fallback.",
        )

        # Begrenzung der generierten Antwortlänge. Hilft besonders bei lokalen
        # Modellen und langsamen Gateways gegen unnötige Timeouts.
        ANSWER_SYNTHESIS_MAX_TOKENS: int = Field(
            default=700,
            ge=64,
            le=4096,
            description="Maximale Ausgabetokens für den optionalen Antwort-Synthese-Fallback.",
        )

        # ------------------------------------------------------------------
        # 5) HTTP- und Laufzeitverhalten
        # ------------------------------------------------------------------
        # Gesamt-Timeout für den Proxy-Call. RAG kann durch Embedding-Suche,
        # Re-Ranking und LLM-Generierung länger dauern. 120 bis 180 Sekunden sind
        # für größere Dokumentbestände realistisch.
        REQUEST_TIMEOUT_SECONDS: float = Field(
            default=180.0,
            ge=5.0,
            le=600.0,
            description="Gesamtes HTTP-Timeout für den Proxy-Aufruf.",
        )

        # Statusmeldungen im OpenWebUI-Chat. Einschalten, weil Nutzer bei RAG sonst
        # nicht erkennen, ob Retrieval/Generierung noch läuft.
        STATUS_UPDATES: bool = Field(
            default=True,
            description="Zeigt OpenWebUI-Statusmeldungen während der Verarbeitung.",
        )

        # Blendet die finale Statusmeldung aus, falls die OpenWebUI-Version dieses
        # Feld unterstützt. Praktisch für besonders cleane Chats. Für Debugging
        # eher False lassen.
        STATUS_MODE: str = Field(
            default="minimal",
            description="Status-UX: off, minimal oder verbose. Minimal vermeidet lange Statuslisten über der Antwort.",
        )

        HIDE_FINAL_STATUS: bool = Field(
            default=False,
            description="Blendet die finale Statusmeldung nach Abschluss aus, sofern OpenWebUI dies unterstützt.",
        )

        # ------------------------------------------------------------------
        # 6) Quellen-UX in OpenWebUI
        # ------------------------------------------------------------------
        # Native Citation-Events erzeugen klickbare Quellen-/Citation-Elemente in
        # OpenWebUI. Das ist die professionellste UX und sollte aktiv sein.
        EMIT_CITATION_EVENTS: bool = Field(
            default=True,
            description="Sendet Quellen als native OpenWebUI-Citation-Events.",
        )

        # Legacy-Source-Events nur aktivieren, wenn deine OpenWebUI-Version oder
        # ein Custom-Frontend keine Citation-Events verarbeitet. Normalerweise
        # deaktiviert lassen, um doppelte Quellenanzeigen zu vermeiden.
        EMIT_LEGACY_SOURCE_EVENTS: bool = Field(
            default=False,
            description="Sendet zusätzlich Legacy-Source-Events für ältere OpenWebUI-Setups.",
        )

        # Begrenzung der Event-Anzahl. Zu viele Events machen den Chat schwer
        # lesbar und können Frontends verlangsamen. 10 bis 20 sind meist ideal.
        MAX_SOURCE_EVENTS: int = Field(
            default=20,
            ge=1,
            le=MAX_SOURCE_EVENTS_HARD_LIMIT,
            description="Maximale Anzahl nativer Quellen-/Citation-Events.",
        )

        # Zeigt numerische Relevanzwerte, sofern RAGFlow/Proxy Scores liefert.
        # Der Audit-Standard zeigt auch bei False ein grobes Relevanzlabel, aber
        # keine Prozentwerte.
        SHOW_SOURCE_SCORES: bool = Field(
            default=False,
            description="Zeigt numerische Relevanzwerte in Metadaten und Quellenübersicht.",
        )

        # Debug-Metadaten in Quellen. Nur für Admins. Kann interne IDs, Dataset-
        # oder Chunk-Informationen sichtbar machen.
        SHOW_SOURCE_DEBUG: bool = Field(
            default=False,
            description="Zeigt technische IDs wie Chunk-, Dokument- und Dataset-ID in Debug-Ausgaben.",
        )

        # Zusätzlicher Quellenblock im Antworttext:
        #   none     keine Markdown-Quellenübersicht; nur native Citations
        #   audit    Nachweistabelle mit Quellen-ID, Fundstelle, Relevanz und Link
        #   compact  kurze Tabelle
        #   detailed ausführliche Quellen mit Snippets; gut für Reviews/Tests
        SOURCE_MARKDOWN_MODE: str = Field(
            default="audit",
            description="Zusätzliche Quellenübersicht im Antworttext: none, compact, detailed oder audit.",
        )

        # Hängt die Markdown-Quellenübersicht an den Antworttext an. Wenn OpenWebUI
        # Citations optisch sehr gut darstellt, kann man das auf False setzen, um
        # den Chat noch cleaner zu machen.
        APPEND_SOURCE_OVERVIEW: bool = Field(
            default=True,
            description="Hängt eine auditierbare Nachweistabelle an die Antwort an.",
        )

        SHOW_LOCATOR_QUALITY: bool = Field(
            default=True,
            description="Zeigt im Audit-Modus, wenn eine Fundstelle nur grob bestimmbar ist.",
        )

        # Sicherheitsventil für Quellenlinks. Standardmäßig werden URLs, die auf
        # den Connector-Proxy selbst zeigen, NICHT als Quellenlink verwendet. Das
        # verhindert irreführende Links auf Backend-/API-Endpunkte. Aktivieren nur,
        # wenn dein Connector bewusst echte Preview-Seiten unter seiner eigenen
        # Domain ausliefert.
        ALLOW_CONNECTOR_SOURCE_LINKS: bool = Field(
            default=False,
            description="Erlaubt Quellenlinks auf die Connector-Proxy-Domain. Standardmäßig deaktiviert, damit keine Backend-URLs als Preview erscheinen.",
        )

        # Entfernt technische Referenzmarker, die manche RAG-Engines in Antworten
        # einbetten, z. B. {{source:0}}, ##0$$ oder [ID:0]. Im Normalbetrieb True.
        CLEAN_RAGFLOW_MARKERS: bool = Field(
            default=True,
            description="Entfernt rohe RAGFlow-Referenzmarker wie {{source:0}}, ##0$$ oder [ID:0].",
        )

        # Fehlerausgaben:
        #   safe  Endnutzerfreundliche Fehler ohne technische Response-Bodies
        #   debug Ergänzt gekürzte Proxy-/Fallback-Antworten zur Fehlersuche
        # Debug nie dauerhaft aktiv lassen, wenn sensible Backend-Fehler möglich
        # sind.
        ERROR_VERBOSITY: str = Field(
            default="safe",
            description="safe oder debug. Debug zeigt etwas mehr technische Hinweise, aber keine Secrets.",
        )

    # ------------------------------------------------------------------
    # OpenWebUI-Lifecycle
    # ------------------------------------------------------------------
    # __init__ wird von OpenWebUI beim Laden der Pipe ausgeführt.
    # self.type="manifold" erlaubt, dass die Pipe ein Modell über pipes()
    # bereitstellt. self.citation=False verhindert, dass OpenWebUI eigene
    # automatische Quellenlogik über unsere kontrollierten Citation-Events legt.
    def __init__(self) -> None:
        self.valves = self.Valves()
        self.type = "manifold"
        self.citation = False

    def pipes(self) -> list[dict[str, str]]:
        model_id = _clean(self.valves.MODEL_ID) or DEFAULT_MODEL_ID
        return [
            {
                "id": model_id,
                "name": _clean(self.valves.MODEL_NAME) or DEFAULT_MODEL_NAME,
            }
        ]

    async def pipe(
        self,
        body: dict,
        __event_emitter__=None,
        __user__: Optional[dict] = None,
        __task__: Optional[str] = None,
        __task_body__: Optional[dict] = None,
        __metadata__: Optional[dict] = None,
    ) -> str:
        task = _normalize_task(__task__ or (__metadata__ or {}).get("task"))
        if task:
            return _task_response(task, __task_body__ or body)

        started_at = time.perf_counter()
        config_error = _configuration_error(self.valves)
        if config_error:
            await _emit_status(self.valves, __event_emitter__, config_error, done=True)
            return _error_card("Konfiguration unvollständig", config_error)

        await _emit_status(
            self.valves,
            __event_emitter__,
            "RAGFlow-Dataset wird vorbereitet …",
            done=False,
        )

        url = _proxy_chat_url(self.valves.CONNECTOR_PROXY_BASE_URL)
        payload = _build_payload(self.valves, body, __user__)
        headers = _build_headers(self.valves.CONNECTOR_PROXY_SHARED_SECRET)

        try:
            await _emit_status(
                self.valves,
                __event_emitter__,
                "Anfrage wird an den RAGFlow-Proxy übergeben …",
                done=False,
            )
            data = await _post_json(
                url,
                payload,
                headers,
                verify_ssl=self.valves.CONNECTOR_PROXY_VERIFY_SSL,
                ca_bundle=self.valves.CONNECTOR_PROXY_CA_BUNDLE,
                timeout_seconds=self.valves.REQUEST_TIMEOUT_SECONDS,
                tls_debug=self.valves.TLS_DEBUG,
                error_verbosity=self.valves.ERROR_VERBOSITY,
            )
        except PipeError as exc:
            await _emit_status(self.valves, __event_emitter__, exc.status, done=True)
            return _error_card("RAGFlow-Anfrage fehlgeschlagen", exc.user_message)

        await _emit_status(
            self.valves,
            __event_emitter__,
            "Antwort empfangen, Quellen werden geprüft …",
            done=False,
        )

        sources = _normalize_sources(
            _extract_sources(data),
            show_scores=self.valves.SHOW_SOURCE_SCORES,
            show_debug=self.valves.SHOW_SOURCE_DEBUG,
            limit=self.valves.MAX_SOURCE_EVENTS,
            connector_base_url=self.valves.CONNECTOR_PROXY_BASE_URL,
            allow_connector_source_links=self.valves.ALLOW_CONNECTOR_SOURCE_LINKS,
        )

        raw_answer = _extract_answer(data)
        if _looks_like_source_markdown_only(raw_answer):
            raw_answer = ""
        else:
            raw_answer = _strip_appended_source_markdown(raw_answer)
        raw_answer = _annotate_answer_source_markers(raw_answer, sources)
        answer = _clean_answer(
            raw_answer,
            clean_ragflow_markers=self.valves.CLEAN_RAGFLOW_MARKERS,
        )

        question = _last_user_text((body if isinstance(body, Mapping) else {}).get("messages") or [])

        if not answer and sources and _is_source_inventory_question(question):
            answer = _source_inventory_answer(sources)

        if not answer and sources and bool(self.valves.ENABLE_ANSWER_SYNTHESIS_FALLBACK):
            await _emit_status(
                self.valves,
                __event_emitter__,
                "Antwort wird aus Dokumentkontext formuliert …",
                done=False,
                hidden=True,
            )
            try:
                answer = await _synthesize_answer_from_sources(
                    self.valves,
                    body,
                    sources,
                    verify_ssl=self.valves.CONNECTOR_PROXY_VERIFY_SSL,
                    ca_bundle=self.valves.CONNECTOR_PROXY_CA_BUNDLE,
                    tls_debug=self.valves.TLS_DEBUG,
                    error_verbosity=self.valves.ERROR_VERBOSITY,
                )
                answer = _postprocess_synthesized_answer(answer, sources, question)
            except PipeError as exc:
                LOGGER.warning("Answer synthesis fallback failed: %s", exc.status)

        if self.valves.EMIT_CITATION_EVENTS:
            for source in sources:
                await _emit_event(__event_emitter__, _citation_event(source))

        if self.valves.EMIT_LEGACY_SOURCE_EVENTS:
            for source in sources:
                await _emit_event(__event_emitter__, {"type": "source", "data": source})

        elapsed = max(0.0, time.perf_counter() - started_at)
        source_group_count = len(_group_sources(sources)) if sources else 0
        await _emit_status(
            self.valves,
            __event_emitter__,
            _completion_status(
                source_group_count,
                elapsed,
                generated=bool(answer),
                hit_count=len(sources),
            ),
            done=True,
            hidden=self.valves.HIDE_FINAL_STATUS,
        )

        return _compose_final_answer(answer, sources, self.valves)


# =============================================================================
# KONFIGURATIONSPRÜFUNG
# =============================================================================
# Alle kritischen Admin-Fehler werden vor dem ersten Netzwerkaufruf abgefangen.
# Dadurch sieht der Nutzer im Chat eine klare Meldung statt eines Tracebacks.
def _configuration_error(valves: Pipe.Valves) -> str:
    """Validate administrator-provided Valves before network calls."""
    base_url = _clean(valves.CONNECTOR_PROXY_BASE_URL)
    if not base_url:
        return "Fehlender Wert: `CONNECTOR_PROXY_BASE_URL`."

    parsed = urlsplit(base_url)
    if parsed.scheme not in {"http", "https"} or not parsed.netloc:
        return "Ungültiger Wert: `CONNECTOR_PROXY_BASE_URL` muss eine vollständige HTTP(S)-URL sein."

    if not _clean(valves.CONNECTOR_PROXY_SHARED_SECRET):
        return "Fehlender Wert: `CONNECTOR_PROXY_SHARED_SECRET`."

    markdown_mode = _source_markdown_mode(valves.SOURCE_MARKDOWN_MODE)
    if markdown_mode not in {"none", "compact", "detailed", "audit"}:
        return "Ungültiger Wert: `SOURCE_MARKDOWN_MODE` muss `none`, `compact`, `detailed` oder `audit` sein."

    error_verbosity = _clean(valves.ERROR_VERBOSITY).lower() or "safe"
    if error_verbosity not in VALID_ERROR_VERBOSITY:
        return "Ungültiger Wert: `ERROR_VERBOSITY` muss `safe` oder `debug` sein."

    retrieval_fallback = _clean(valves.RETRIEVAL_ONLY_FALLBACK).lower() or "brief"
    if retrieval_fallback not in VALID_RETRIEVAL_ONLY_FALLBACK:
        return "Ungültiger Wert: `RETRIEVAL_ONLY_FALLBACK` muss `diagnostic`, `brief` oder `sources_only` sein."

    status_mode = _clean(getattr(valves, "STATUS_MODE", "minimal")).lower() or "minimal"
    if status_mode not in VALID_STATUS_MODES:
        return "Ungültiger Wert: `STATUS_MODE` muss `off`, `minimal` oder `verbose` sein."

    if bool(valves.ENABLE_ANSWER_SYNTHESIS_FALLBACK):
        answer_base_url = _clean(valves.ANSWER_LLM_BASE_URL)
        if not answer_base_url:
            return "Fehlender Wert: `ANSWER_LLM_BASE_URL` für den Antwort-Synthese-Fallback."
        parsed_answer_url = urlsplit(answer_base_url)
        if parsed_answer_url.scheme not in {"http", "https"} or not parsed_answer_url.netloc:
            return "Ungültiger Wert: `ANSWER_LLM_BASE_URL` muss eine vollständige HTTP(S)-URL sein."
        if not _clean(valves.ANSWER_LLM_API_KEY):
            return "Fehlender Wert: `ANSWER_LLM_API_KEY` für den Antwort-Synthese-Fallback."
        if not _clean(valves.ANSWER_LLM_MODEL):
            return "Fehlender Wert: `ANSWER_LLM_MODEL` für den Antwort-Synthese-Fallback."

    ca_bundle = _clean(valves.CONNECTOR_PROXY_CA_BUNDLE)
    if bool(valves.CONNECTOR_PROXY_VERIFY_SSL) and ca_bundle:
        path = Path(ca_bundle)
        if not path.exists():
            return "Ungültiger Wert: Das angegebene CA-Bundle existiert nicht."
        if not path.is_file():
            return "Ungültiger Wert: Das angegebene CA-Bundle ist keine Datei."

    return ""


# =============================================================================
# PAYLOAD-AUFBAU FÜR DEN CONNECTOR-PROXY
# =============================================================================
# Hier wird aus OpenWebUIs Chat-Body ein Proxy-Request gebaut.
# Entscheidend ist: Die Pipe sendet nicht nur messages und top_k, sondern auch
# explizite Chat-RAG-Hinweise, damit der Proxy nicht bloß Suchtreffer liefert.
def _build_payload(valves: Pipe.Valves, body: dict, user: Optional[dict]) -> dict[str, Any]:
    """Build the JSON request sent to the Connector Proxy."""
    request_body = body if isinstance(body, Mapping) else {}
    raw_messages = request_body.get("messages")
    messages = raw_messages if isinstance(raw_messages, list) else []
    messages = _messages_with_rag_system_prompt(
        messages,
        system_prompt=valves.RAG_SYSTEM_PROMPT,
        enabled=bool(valves.INJECT_RAG_SYSTEM_PROMPT),
    )

    model = (
        _clean(request_body.get("model"))
        or _clean(valves.MODEL_ID)
        or DEFAULT_MODEL_ID
    )

    user_payload: dict[str, Any] = {}
    if isinstance(user, Mapping):
        user_payload = _drop_empty(
            {
                "id": user.get("id"),
                "email": user.get("email"),
                "name": user.get("name"),
                "role": user.get("role"),
            }
        )

    payload = {
        "artifact_id": valves.ARTIFACT_ID,
        "dataset_id": valves.DATASET_ID,
        "chat_id": valves.RAGFLOW_CHAT_ID,
        "messages": messages,
        "model": model,
        "top_k": int(valves.TOP_K),
        "stream": False,
        "user": user_payload,
        "openwebui": {
            "source_events": bool(valves.EMIT_CITATION_EVENTS or valves.EMIT_LEGACY_SOURCE_EVENTS),
            "source_markdown_mode": _source_markdown_mode(valves.SOURCE_MARKDOWN_MODE),
            "source_display_mode": _source_markdown_mode(valves.SOURCE_MARKDOWN_MODE),
            "audit_evidence": _source_markdown_mode(valves.SOURCE_MARKDOWN_MODE) == "audit",
            "expects_generated_answer": True,
        },
    }

    if bool(valves.SEND_GENERATIVE_RAG_HINTS):
        # Diese Felder sind für den eigenen Connector-Proxy gedacht. Sie machen den
        # Unterschied zwischen reiner Suche und echter Chat-RAG-Antwort explizit.
        payload.update(
            {
                "mode": "chat",
                "query_mode": "chat",
                "response_mode": "chat",
                "generate_answer": True,
                "return_answer": True,
                "return_sources": True,
                "retrieval_only": False,
                "extra_body": {
                    "reference": True,
                    "reference_metadata": {
                        "chunks": True,
                        "doc_aggs": True,
                        "document": True,
                        "positions": True,
                    },
                },
                "ragflow": {
                    "mode": "chat",
                    "generate_answer": True,
                    "retrieval_only": False,
                    "conversation": True,
                    "requires_answer_field": True,
                },
            }
        )

    return _drop_empty(payload)



def _messages_with_rag_system_prompt(
    messages: list[Any],
    *,
    system_prompt: str,
    enabled: bool,
) -> list[dict[str, Any]]:
    """Inject the RAG behavior prompt once without duplicating it."""
    normalized_messages = [dict(message) for message in messages if isinstance(message, Mapping)]
    prompt = str(system_prompt or "").strip()
    if not enabled or not prompt:
        return normalized_messages

    marker = "RAG_ASSISTANT_BEHAVIOR"
    prompt_with_marker = f"<!-- {marker} -->\n{prompt}"

    if normalized_messages and normalized_messages[0].get("role") == "system":
        existing = str(normalized_messages[0].get("content") or "").strip()
        if marker in existing:
            return normalized_messages
        merged = f"{prompt_with_marker}\n\n{existing}" if existing else prompt_with_marker
        normalized_messages[0] = {**normalized_messages[0], "content": merged}
        return normalized_messages

    return [{"role": "system", "content": prompt_with_marker}, *normalized_messages]

def _build_headers(shared_secret: str) -> dict[str, str]:
    """Create HTTP headers without exposing the secret anywhere else."""
    return {
        "Authorization": f"Bearer {_clean(shared_secret)}",
        "Content-Type": "application/json",
        "Accept": "application/json",
        "User-Agent": "OpenWebUI-RAGFlow-Dataset-Pipe/3.4",
    }


def _proxy_chat_url(base_url: str) -> str:
    """Resolve the final Connector Proxy chat endpoint."""
    return _clean(base_url).rstrip("/") + PROXY_CHAT_PATH


# =============================================================================
# HTTP-KOMMUNIKATION UND FEHLERBEHANDLUNG
# =============================================================================
# Netzwerkfehler werden in PipeError übersetzt, damit im Chat kontrollierte,
# sichere und verständliche Fehlermeldungen erscheinen.
async def _post_json(
    url: str,
    payload: dict[str, Any],
    headers: dict[str, str],
    *,
    verify_ssl: bool,
    ca_bundle: str,
    timeout_seconds: float,
    tls_debug: bool,
    error_verbosity: str,
) -> dict[str, Any]:
    try:
        verify = _httpx_verify(verify_ssl, ca_bundle)
    except ValueError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Die TLS-Konfiguration der Pipe ist ungültig. Prüfe CA-Bundle, Zertifikatskette und `CONNECTOR_PROXY_VERIFY_SSL`.",
            status="TLS-Konfiguration ungültig.",
        ) from exc

    timeout = _httpx_timeout(timeout_seconds)

    try:
        async with httpx.AsyncClient(
            timeout=timeout,
            verify=verify,
            follow_redirects=False,
        ) as client:
            response = await client.post(url, json=payload, headers=headers)
            response.raise_for_status()
    except httpx.TimeoutException as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "RAGFlow hat innerhalb des konfigurierten Zeitfensters nicht geantwortet.",
            status="RAGFlow-Timeout.",
        ) from exc
    except httpx.HTTPStatusError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        status_code = exc.response.status_code
        raise PipeError(
            _http_status_user_message(status_code, exc.response, error_verbosity),
            status=f"RAGFlow-Proxy antwortete mit HTTP {status_code}.",
        ) from exc
    except httpx.ConnectError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Der RAGFlow-Proxy war nicht erreichbar. Prüfe Proxy-URL, Netzwerk, DNS und TLS.",
            status="RAGFlow-Proxy nicht erreichbar.",
        ) from exc
    except httpx.RequestError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Die Anfrage an den RAGFlow-Proxy ist fehlgeschlagen.",
            status="RAGFlow-Proxy-Anfrage fehlgeschlagen.",
        ) from exc
    except Exception as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Die RAGFlow-Antwort konnte wegen eines unerwarteten Fehlers nicht erzeugt werden.",
            status="RAGFlow-Antwort fehlgeschlagen.",
        ) from exc

    try:
        data = response.json()
    except ValueError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Der RAGFlow-Proxy lieferte keine gültige JSON-Antwort.",
            status="Ungültige JSON-Antwort.",
        ) from exc

    if not isinstance(data, Mapping):
        raise PipeError(
            "Der RAGFlow-Proxy lieferte ein unerwartetes Antwortformat.",
            status="Unerwartetes Antwortformat.",
        )

    return dict(data)



# =============================================================================
# OPTIONALER ANTWORT-SYNTHESE-FALLBACK
# =============================================================================
# Dieser Block ist nur relevant, wenn der Proxy keinen KI-Text erzeugt.
# Dann nutzt die Pipe Quellen als Kontext und ruft ein OpenAI-kompatibles LLM auf.
async def _synthesize_answer_from_sources(
    valves: Pipe.Valves,
    body: dict,
    sources: list[dict[str, Any]],
    *,
    verify_ssl: bool,
    ca_bundle: str,
    tls_debug: bool,
    error_verbosity: str,
) -> str:
    original_messages = (body if isinstance(body, Mapping) else {}).get("messages") or []
    question = _last_user_text(original_messages)
    context = _sources_context(sources, max_chars=int(valves.ANSWER_CONTEXT_MAX_CHARS))
    if not question or not context:
        return ""

    url = _openai_chat_completions_url(valves.ANSWER_LLM_BASE_URL)
    headers = {
        "Authorization": f"Bearer {_clean(valves.ANSWER_LLM_API_KEY)}",
        "Content-Type": "application/json",
        "Accept": "application/json",
        "User-Agent": "OpenWebUI-RAGFlow-Dataset-Pipe/3.4 Answer-Synthesis",
    }
    payload = {
        "model": _clean(valves.ANSWER_LLM_MODEL),
        "messages": [
            {"role": "system", "content": DEFAULT_SYNTHESIS_SYSTEM_PROMPT},
            *_recent_messages_for_synthesis(original_messages, max_messages=6),
            {
                "role": "user",
                "content": (
                    "Beantworte die letzte Nutzerfrage anhand dieses Dokumentkontexts. "
                    "Der Kontext ist nicht selbst die Antwort, sondern nur Evidenz.\n\n"
                    "Dokumentkontext:\n"
                    f"{context}\n\n"
                    "Formuliere jetzt die finale Chat-Antwort. Belege jede wesentliche Aussage mit den "
                    "Quellenmarken [S1], [S2] usw. aus dem Kontext. Keine Quellenliste und keine "
                    "Dateinamen-Aufzählung, außer die Nutzerfrage verlangt ausdrücklich eine Quellen-/Dokumentliste."
                ),
            },
        ],
        "temperature": float(valves.ANSWER_SYNTHESIS_TEMPERATURE),
        "max_tokens": int(valves.ANSWER_SYNTHESIS_MAX_TOKENS),
        "stream": False,
    }

    data = await _post_answer_llm_json(
        url,
        _drop_empty(payload),
        headers,
        verify_ssl=verify_ssl,
        ca_bundle=ca_bundle,
        timeout_seconds=valves.REQUEST_TIMEOUT_SECONDS,
        tls_debug=tls_debug,
        error_verbosity=error_verbosity,
    )
    answer = _clean_answer(
        _strip_appended_source_markdown(_extract_answer(data)),
        clean_ragflow_markers=valves.CLEAN_RAGFLOW_MARKERS,
    )
    if not answer:
        raise PipeError(
            "Der Antwort-Synthese-Fallback lieferte keinen verwertbaren Antworttext.",
            status="Antwort-Synthese ohne Inhalt.",
        )
    return answer


async def _post_answer_llm_json(
    url: str,
    payload: dict[str, Any],
    headers: dict[str, str],
    *,
    verify_ssl: bool,
    ca_bundle: str,
    timeout_seconds: float,
    tls_debug: bool,
    error_verbosity: str,
) -> dict[str, Any]:
    try:
        verify = _httpx_verify(verify_ssl, ca_bundle)
    except ValueError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Die TLS-Konfiguration für den Antwort-Synthese-Fallback ist ungültig.",
            status="TLS-Konfiguration für Antwort-Synthese ungültig.",
        ) from exc

    try:
        async with httpx.AsyncClient(
            timeout=_httpx_timeout(timeout_seconds),
            verify=verify,
            follow_redirects=False,
        ) as client:
            response = await client.post(url, json=payload, headers=headers)
            response.raise_for_status()
    except httpx.TimeoutException as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Der Antwort-Synthese-Fallback hat nicht rechtzeitig geantwortet.",
            status="Antwort-Synthese-Timeout.",
        ) from exc
    except httpx.HTTPStatusError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        status_code = exc.response.status_code
        raise PipeError(
            _answer_llm_status_message(status_code, exc.response, error_verbosity),
            status=f"Antwort-Synthese antwortete mit HTTP {status_code}.",
        ) from exc
    except httpx.RequestError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Der Antwort-Synthese-Fallback war nicht erreichbar.",
            status="Antwort-Synthese nicht erreichbar.",
        ) from exc

    try:
        data = response.json()
    except ValueError as exc:
        _log_proxy_error(exc, url, ca_bundle, debug=tls_debug)
        raise PipeError(
            "Der Antwort-Synthese-Fallback lieferte keine gültige JSON-Antwort.",
            status="Ungültige JSON-Antwort der Antwort-Synthese.",
        ) from exc

    if not isinstance(data, Mapping):
        raise PipeError(
            "Der Antwort-Synthese-Fallback lieferte ein unerwartetes Antwortformat.",
            status="Unerwartetes Antwort-Synthese-Format.",
        )
    return dict(data)


def _answer_llm_status_message(status_code: int, response: httpx.Response, error_verbosity: str) -> str:
    if status_code in {401, 403}:
        message = "Der Antwort-Synthese-Fallback hat die Anfrage abgelehnt. Prüfe `ANSWER_LLM_API_KEY` und Berechtigungen."
    elif status_code == 404:
        message = "Der Antwort-Synthese-Endpunkt wurde nicht gefunden. Prüfe `ANSWER_LLM_BASE_URL`."
    elif status_code == 429:
        message = "Der Antwort-Synthese-Fallback ist ausgelastet oder limitiert Anfragen."
    elif 500 <= status_code <= 599:
        message = f"Der Antwort-Synthese-Fallback meldete einen Serverfehler (HTTP {status_code})."
    else:
        message = f"Der Antwort-Synthese-Fallback antwortete mit HTTP {status_code}."

    if _clean(error_verbosity).lower() == "debug":
        body_hint = _compact(_clean(response.text), 500)
        if body_hint:
            message += f"\n\n**Fallback-Hinweis:** `{_escape_inline_code(body_hint)}`"
    return message


def _openai_chat_completions_url(base_url: str) -> str:
    clean = _clean(base_url).rstrip("/")
    parsed = urlsplit(clean)
    path = parsed.path.rstrip("/")
    if path.endswith("/chat/completions"):
        return clean
    if path.endswith("/v1") or path.endswith("/api"):
        return clean + "/chat/completions"
    return clean + "/v1/chat/completions"




def _recent_messages_for_synthesis(messages: Any, *, max_messages: int = 6) -> list[dict[str, str]]:
    """Return a compact user/assistant history for answer synthesis."""
    if not isinstance(messages, Sequence) or isinstance(messages, (str, bytes)):
        return []

    normalized: list[dict[str, str]] = []
    for message in messages:
        if not isinstance(message, Mapping):
            continue
        role = _clean(message.get("role")).lower()
        if role not in {"user", "assistant"}:
            continue
        content = message.get("content")
        text = content if isinstance(content, str) else _content_parts_to_text(content)
        text = _compact(_clean_answer(text, clean_ragflow_markers=True), 1800)
        if text:
            normalized.append({"role": role, "content": text})

    return normalized[-max(1, int(max_messages or 6)):]


def _is_source_inventory_question(question: Any) -> bool:
    text = _clean(question).lower()
    if not text:
        return False
    return bool(
        re.search(r"\b(welche|welche\s+dateien|zeige|liste|nenn[e]?)\b.*\b(quellen|dokumente|dateien)\b", text)
        or re.search(r"\b(quellen|dokumente|dateien)\b.*\b(verfügbar|vorhanden|gefunden|angebunden|indexiert)\b", text)
        or re.search(r"\bwas\s+(ist|sind)\s+.*\b(quellen|dokumente|dateien)\b", text)
    )


def _source_inventory_answer(sources: list[dict[str, Any]]) -> str:
    groups = _group_sources(sources)
    if not groups:
        return "Ich habe aktuell keine Quellen gefunden."

    lines = ["Ich habe folgende Dokumente als Quellen gefunden:", ""]
    for index, group in enumerate(groups[:12], start=1):
        source = group[0]
        metadata = _metadata(source)
        title = _clean(source.get("name") or metadata.get("document_name") or metadata.get("doc_name") or "Quelle")
        location = _location(metadata)
        details = []
        if location and location != "Fundstelle nicht angegeben":
            details.append(location)
        if len(group) > 1:
            details.append(f"{len(group)} Treffer")
        suffix = f" — {' · '.join(details)}" if details else ""
        lines.append(f"{index}. **{_escape_markdown(title)}**{suffix}")

    if len(groups) > 12:
        lines.append(f"\nWeitere {len(groups) - 12} Quellen wurden ausgeblendet.")

    lines.append("\nStelle mir eine konkrete Frage zum Inhalt dieser Dokumente, dann formuliere ich daraus eine inhaltliche RAG-Antwort.")
    return "\n".join(lines).strip()


def _postprocess_synthesized_answer(answer: str, sources: list[dict[str, Any]], question: Any) -> str:
    text = _clean_answer(answer, clean_ragflow_markers=True)
    if not text:
        return ""

    text = re.sub(r"\bDie\s+Verfügbaren\s+Quellen\b", "Die verfügbaren Quellen", text)
    text = re.sub(r"\s*\[(?:Quelle|Source)\s*\d+\]", "", text, flags=re.IGNORECASE)
    text = re.sub(r"\s*\[\d+\]", "", text)
    text = re.sub(r" {2,}", " ", text)

    if not _is_source_inventory_question(question) and _looks_like_source_inventory_answer(text, sources):
        return ""

    return text.strip()


def _annotate_answer_source_markers(answer: str, sources: list[dict[str, Any]]) -> str:
    if not answer or not sources:
        return answer

    def source_id_for_index(raw_index: str) -> str:
        index = _int_or_none(raw_index)
        if index is None or index < 0 or index >= len(sources):
            return f"[ID:{raw_index}]"
        metadata = _metadata(sources[index])
        return f"[{_source_id(sources[index], metadata)}]"

    text = re.sub(
        r"\[\s*ID\s*:\s*(\d+)\s*\]",
        lambda match: source_id_for_index(match.group(1)),
        str(answer),
        flags=re.IGNORECASE,
    )
    text = re.sub(
        r"\{\{\s*source\s*:\s*(\d+)\s*\}\}",
        lambda match: source_id_for_index(match.group(1)),
        text,
        flags=re.IGNORECASE,
    )
    text = re.sub(
        r"##(\d+)\$\$",
        lambda match: source_id_for_index(match.group(1)),
        text,
    )
    return text


def _looks_like_source_inventory_answer(answer: str, sources: list[dict[str, Any]]) -> bool:
    clean_answer = _clean(answer).lower()
    if not clean_answer:
        return False

    inventory_phrases = (
        "verfügbaren quellen",
        "gefundenen quellen",
        "folgende quellen",
        "quellen sind",
        "dokumente sind",
    )
    if any(phrase in clean_answer for phrase in inventory_phrases):
        title_hits = 0
        for source in sources:
            title = _clean(source.get("name") or _metadata(source).get("document_name") or "").lower()
            if title and title in clean_answer:
                title_hits += 1
        return title_hits >= 2 or len(clean_answer) < 700

    return False


def _looks_like_source_markdown_only(text: str) -> bool:
    raw = str(text or "").strip()
    if not raw:
        return False
    start = _source_markdown_start(raw)
    if start is None:
        return _source_table_density(raw) >= 0.55
    prefix = raw[:start].strip()
    return not _has_substantive_answer_prefix(prefix)


def _contains_source_markdown(text: str) -> bool:
    raw = str(text or "")
    if not raw.strip():
        return False
    return _source_markdown_start(raw) is not None or _source_table_density(raw) >= 0.55


def _strip_appended_source_markdown(text: str) -> str:
    raw = str(text or "")
    start = _source_markdown_start(raw)
    if start is None:
        return raw.strip()
    prefix = raw[:start].strip()
    if not _has_substantive_answer_prefix(prefix):
        return raw.strip()
    return prefix


def _source_markdown_start(text: str) -> Optional[int]:
    offset = 0
    previous_nonempty: tuple[int, str] | None = None
    for line in str(text or "").replace("\r\n", "\n").replace("\r", "\n").splitlines(keepends=True):
        line_text = line.rstrip("\n")
        stripped = line_text.strip()
        if _is_source_markdown_heading(stripped) or _is_source_table_header(stripped):
            if previous_nonempty and previous_nonempty[1] == "---":
                return previous_nonempty[0]
            return offset
        if stripped:
            previous_nonempty = (offset, stripped)
        offset += len(line)
    return None


def _is_source_markdown_heading(line: str) -> bool:
    heading = line.lstrip("#").strip().lower()
    return heading in {
        "nachweise",
        "quellen",
        "quellenüberblick",
        "gefundene quellen",
        "found sources",
        "sources",
        "source overview",
    }


def _is_source_table_header(line: str) -> bool:
    compact = " ".join(line.lower().split())
    return (
        compact.startswith("|")
        and "| id |" in compact
        and "| dokument |" in compact
        and "| fundstelle |" in compact
        and "| öffnen |" in compact
    )


def _source_table_density(text: str) -> float:
    lines = [line.strip() for line in str(text or "").splitlines() if line.strip()]
    if not lines:
        return 0.0
    source_lines = 0
    for line in lines:
        lower = line.lower()
        if _is_source_table_header(line) or lower.startswith("|---"):
            source_lines += 1
        elif lower.startswith("| s") and "|" in lower:
            source_lines += 1
        elif lower.startswith("**nachweisqualität:**"):
            source_lines += 1
    return source_lines / len(lines)


def _has_substantive_answer_prefix(prefix: str) -> bool:
    clean = _clean(prefix.replace("---", " "))
    if not clean:
        return False
    if _looks_like_source_markdown_only(clean):
        return False
    return bool(re.search(r"[0-9A-Za-zÄÖÜäöüß]{2,}", clean))


def _is_low_information_snippet(snippet: Any) -> bool:
    text = _clean(snippet).lower()
    if not text:
        return True
    if re.fullmatch(r"\d+\s+treffer\s+im\s+dokument", text):
        return True
    if re.fullmatch(r"(seite|page)\s+\d+", text):
        return True
    if len(text) < 30 and not re.search(r"[.!?:;]", text):
        return True
    return False

def _sources_context(sources: list[dict[str, Any]], *, max_chars: int) -> str:
    groups = _group_sources(sources)
    lines: list[str] = []
    remaining = max(1000, int(max_chars or 12000))

    for display_rank, group in enumerate(groups, start=1):
        for source in group[:2]:
            metadata = _metadata(source)
            title = _clean(source.get("name") or metadata.get("document_name") or metadata.get("doc_name") or "Quelle")
            location = _location(metadata)
            snippet = _clean_snippet(source.get("text") or source.get("snippet") or "")
            if not snippet or _is_low_information_snippet(snippet):
                continue
            source_id = _source_id(source, metadata)
            block = f"[{source_id}: {title} · {location}]\n{snippet}\n"
            if len(block) > remaining:
                block = block[: max(0, remaining - 1)].rstrip() + "…"
            if not block.strip():
                break
            lines.append(block)
            remaining -= len(block)
            if remaining <= 0:
                return "\n".join(lines).strip()

    return "\n".join(lines).strip()

def _http_status_user_message(status_code: int, response: httpx.Response, error_verbosity: str) -> str:
    if status_code in {401, 403}:
        message = "RAGFlow-Proxy hat die Anfrage abgelehnt. Prüfe `CONNECTOR_PROXY_SHARED_SECRET` und Proxy-Berechtigungen."
    elif status_code == 404:
        message = "Der RAGFlow-Proxy-Endpunkt wurde nicht gefunden. Prüfe `CONNECTOR_PROXY_BASE_URL` und Proxy-Routing."
    elif status_code == 408:
        message = "Der RAGFlow-Proxy hat die Anfrage wegen Zeitüberschreitung beendet."
    elif status_code == 429:
        message = "Der RAGFlow-Proxy ist ausgelastet oder limitiert Anfragen."
    elif 500 <= status_code <= 599:
        message = f"Der RAGFlow-Proxy meldete einen Serverfehler (HTTP {status_code})."
    else:
        message = f"Der RAGFlow-Proxy antwortete mit HTTP {status_code}."

    if _clean(error_verbosity).lower() == "debug":
        body_hint = _compact(_clean(response.text), 500)
        if body_hint:
            message += f"\n\n**Proxy-Hinweis:** `{_escape_inline_code(body_hint)}`"

    return message


def _httpx_timeout(timeout_seconds: float) -> httpx.Timeout:
    """Create conservative timeouts suitable for RAG workloads."""
    total = max(5.0, min(float(timeout_seconds or 180.0), 600.0))
    connect = min(20.0, total)
    write = min(30.0, total)
    pool = min(20.0, total)
    return httpx.Timeout(total, connect=connect, read=total, write=write, pool=pool)


def _httpx_verify(verify_ssl: bool, ca_bundle: str) -> bool | ssl.SSLContext:
    if not bool(verify_ssl):
        LOGGER.warning(
            "OpenWebUI Pipe -> Connector Proxy uses VERIFY_SSL=false; this is intended only for debug/dev."
        )
        return False

    ca_path = _clean(ca_bundle)
    if not ca_path:
        return True

    path = Path(ca_path)
    if not path.exists():
        raise ValueError(f"CA bundle does not exist: {ca_path}")
    if not path.is_file():
        raise ValueError(f"CA bundle is not a file: {ca_path}")

    try:
        return ssl.create_default_context(cafile=ca_path)
    except OSError as exc:
        raise ValueError(f"CA bundle is not usable: {ca_path}") from exc


def _log_proxy_error(exc: Exception, url: str, ca_bundle: str, *, debug: bool = False) -> None:
    LOGGER.warning(
        "OpenWebUI Pipe -> Connector Proxy failed target=%s error_class=%s ca_bundle_set=%s",
        _safe_url(url),
        _error_class(exc),
        bool(_clean(ca_bundle)),
        exc_info=debug,
    )


def _safe_url(url: str) -> str:
    parsed = urlsplit(str(url or ""))
    if not parsed.scheme or not parsed.netloc:
        return str(url or "").split("?", 1)[0]

    host = parsed.hostname or ""
    if parsed.port is not None:
        host = f"{host}:{parsed.port}"

    return urlunsplit((parsed.scheme, host, parsed.path.rstrip("/") or "", "", ""))


def _error_class(exc: Exception) -> str:
    if isinstance(exc, httpx.TimeoutException):
        return "TIMEOUT"
    if isinstance(exc, httpx.HTTPStatusError):
        return f"HTTP_{exc.response.status_code}"
    if isinstance(exc, httpx.ConnectError):
        text = str(exc).lower()
        if any(token in text for token in ("certificate", "cert", "ssl", "tls")):
            return "CERTIFICATE_VERIFY_FAILED"
        return "CONNECT_ERROR"
    if isinstance(exc, httpx.RequestError):
        return exc.__class__.__name__.upper()
    if isinstance(exc, ValueError):
        return "CONFIGURATION_OR_RESPONSE_ERROR"
    return exc.__class__.__name__.upper()

# Event-Emission ist optional und fehlertolerant. Falls ein Frontend ein Event
# nicht akzeptiert, darf die eigentliche Antwort nicht scheitern.
async def _emit_status(
    valves: Pipe.Valves,
    emitter,
    description: str,
    *,
    done: bool,
    hidden: bool = False,
) -> None:
    if not bool(valves.STATUS_UPDATES):
        return

    mode = _clean(getattr(valves, "STATUS_MODE", "minimal")).lower() or "minimal"
    if mode == "off":
        return

    if mode == "minimal" and not done:
        # OpenWebUI kann Status-Historien als sichtbare Bullet-Liste anzeigen.
        # Deshalb zeigen wir im Minimalmodus nur den ersten ruhigen Startstatus
        # und unterdrücken technische Zwischenschritte.
        if "vorbereitet" not in _clean(description).lower():
            return

    await _emit_event(
        emitter,
        {
            "type": "status",
            "data": {
                "description": description,
                "done": done,
                "hidden": bool(hidden),
            },
        },
    )


async def _emit_event(emitter, event: dict[str, Any]) -> None:
    if not emitter:
        return

    try:
        await emitter(event)
    except Exception:
        LOGGER.warning("OpenWebUI event emission failed", exc_info=True)


def _completion_status(
    source_count: int,
    elapsed_seconds: float,
    *,
    generated: bool = True,
    hit_count: Optional[int] = None,
) -> str:
    if source_count == 0:
        source_text = "keine Quellen"
    elif source_count == 1:
        source_text = "1 Quelle"
    else:
        source_text = f"{source_count} Quellen"
    if hit_count is not None and hit_count > source_count:
        source_text = f"{source_text} · {hit_count} Treffer"
    answer_text = "Antwort generiert" if generated else "nur Quellen erhalten"
    return f"{answer_text} · {source_text} · {elapsed_seconds:.1f}s"


def _normalize_task(task: Any) -> str:
    if not task:
        return ""

    value = str(task).strip().lower()
    if "." in value:
        value = value.rsplit(".", 1)[-1]
    return value


def _task_response(task: str, task_body: Optional[dict]) -> str:
    text = _last_user_text((task_body or {}).get("messages") or [])

    if "title" in task:
        return _compact_text(text, 80) or "RAGFlow"
    if "tags" in task or "follow_up" in task:
        return "[]"
    if "emoji" in task:
        return "📚"
    if "query" in task or "image_prompt" in task:
        return _compact_text(text, 160)

    return ""


def _last_user_text(messages: Any) -> str:
    if not isinstance(messages, Sequence) or isinstance(messages, (str, bytes)):
        return ""

    for message in reversed(messages):
        if not isinstance(message, Mapping) or message.get("role") != "user":
            continue

        content = message.get("content")
        if isinstance(content, str):
            return content

        if isinstance(content, Sequence) and not isinstance(content, (str, bytes)):
            parts = []
            for item in content:
                if isinstance(item, Mapping):
                    parts.append(str(item.get("text") or ""))
                elif isinstance(item, str):
                    parts.append(item)
            return " ".join(part for part in parts if part)

    return ""


def _compact_text(text: Any, max_length: int) -> str:
    clean = _clean(text)
    if len(clean) <= max_length:
        return clean
    return clean[: max_length - 1].rstrip() + "…"

# Proxy-/RAGFlow-Antworten können viele Formen haben. Dieser Abschnitt sucht
# bewusst nach generiertem Assistant-Text und ignoriert Quellencontainer.
def _extract_answer(data: Mapping[str, Any]) -> str:
    """Extract generated assistant text without confusing source chunks for answers.

    RAGFlow and proxy implementations differ widely: some return
    data.answer, others OpenAI-compatible choices[0].message.content, and
    some stream-like payloads are forwarded as text. This extractor prefers
    high-confidence generated-answer paths and deliberately skips source,
    metadata and message-history containers.
    """
    candidates: list[tuple[int, str]] = []
    _collect_answer_candidates(data, candidates, path=(), depth=0)

    cleaned: list[tuple[int, str]] = []
    seen: set[str] = set()
    for score, text in candidates:
        answer = _clean_answer(text, clean_ragflow_markers=True)
        if not _valid_answer_candidate(answer):
            continue
        fingerprint = _clean(answer)[:500]
        if fingerprint in seen:
            continue
        seen.add(fingerprint)
        cleaned.append((score, answer))

    if not cleaned:
        return ""

    cleaned.sort(key=lambda item: (item[0], min(len(item[1]), 4000)), reverse=True)
    return cleaned[0][1]


def _collect_answer_candidates(
    value: Any,
    candidates: list[tuple[int, str]],
    *,
    path: tuple[str, ...],
    depth: int,
) -> None:
    if depth > 10:
        return

    if isinstance(value, str):
        from_blob = _extract_answer_from_text_blob(value)
        if from_blob:
            candidates.append((55, from_blob))
        return

    if isinstance(value, Mapping):
        role = _clean(value.get("role")).lower()

        # OpenAI-compatible response shape.
        choices = value.get("choices")
        if isinstance(choices, Sequence) and not isinstance(choices, (str, bytes)):
            for choice in choices:
                if not isinstance(choice, Mapping):
                    continue
                message = choice.get("message")
                if isinstance(message, Mapping):
                    content = message.get("content")
                    if isinstance(content, str):
                        candidates.append((98, content))
                    elif isinstance(content, Sequence) and not isinstance(content, (str, bytes)):
                        candidates.append((96, _content_parts_to_text(content)))
                delta = choice.get("delta")
                if isinstance(delta, Mapping) and isinstance(delta.get("content"), str):
                    candidates.append((90, delta["content"]))
                if isinstance(choice.get("text"), str):
                    candidates.append((82, choice["text"]))

        message = value.get("message")
        if isinstance(message, Mapping):
            content = message.get("content")
            message_role = _clean(message.get("role")).lower()
            if isinstance(content, str) and (message_role in {"", "assistant"}):
                candidates.append((92, content))
            elif isinstance(content, Sequence) and not isinstance(content, (str, bytes)):
                candidates.append((90, _content_parts_to_text(content)))
        elif isinstance(message, str) and "error" not in _clean(value.get("status")).lower():
            candidates.append((45, message))

        # Direct generated-answer fields. High-confidence keys are preferred.
        for key in ANSWER_KEYS:
            if key not in value:
                continue
            if key in LOW_CONFIDENCE_ANSWER_KEYS and role not in {"", "assistant"} and path[-1:] != ("data",):
                continue
            score = 95 if key in HIGH_CONFIDENCE_ANSWER_KEYS else 62
            if key == "answer":
                score = 100
            _append_answer_value(value.get(key), candidates, score=score, path=(*path, key), depth=depth + 1)

        for key, child in value.items():
            key_text = str(key)
            if key_text in SKIP_RECURSIVE_ANSWER_KEYS:
                continue
            if key_text in ANSWER_KEYS:
                continue
            if isinstance(child, (Mapping, list, tuple, str)):
                _collect_answer_candidates(child, candidates, path=(*path, key_text), depth=depth + 1)

        return

    if isinstance(value, Sequence) and not isinstance(value, (str, bytes)):
        for index, item in enumerate(value):
            _collect_answer_candidates(item, candidates, path=(*path, str(index)), depth=depth + 1)


def _append_answer_value(
    value: Any,
    candidates: list[tuple[int, str]],
    *,
    score: int,
    path: tuple[str, ...],
    depth: int,
) -> None:
    if isinstance(value, str):
        from_blob = _extract_answer_from_text_blob(value)
        candidates.append((score, from_blob or value))
        return

    if isinstance(value, Mapping):
        nested: list[tuple[int, str]] = []
        _collect_answer_candidates(value, nested, path=path, depth=depth)
        for nested_score, text in nested:
            candidates.append((max(score - 5, nested_score), text))
        return

    if isinstance(value, Sequence) and not isinstance(value, (str, bytes)):
        text = _content_parts_to_text(value)
        if text:
            candidates.append((score - 10, text))
        else:
            _collect_answer_candidates(value, candidates, path=path, depth=depth)


def _extract_answer_from_text_blob(text: str) -> str:
    raw = str(text or "").strip()
    if not raw:
        return ""

    # Some proxies forward SSE as one string: data: {...}\n\ndata: [DONE]
    if raw.startswith("data:") or "\ndata:" in raw:
        pieces: list[str] = []
        for line in raw.splitlines():
            line = line.strip()
            if not line.startswith("data:"):
                continue
            payload = line[5:].strip()
            if not payload or payload == "[DONE]":
                continue
            parsed = _json_loads_safely(payload)
            if parsed is None:
                continue
            if isinstance(parsed, Mapping):
                piece = _stream_delta_text(parsed) or _extract_answer(parsed)
            elif isinstance(parsed, str):
                piece = parsed
            else:
                piece = ""
            if piece:
                pieces.append(piece)
        return _merge_stream_pieces(pieces)

    # Some proxies double-encode a JSON response as a string.
    if raw[:1] in {"{", "["}:
        parsed = _json_loads_safely(raw)
        if isinstance(parsed, Mapping):
            return _extract_answer(parsed)
        if isinstance(parsed, Sequence) and not isinstance(parsed, (str, bytes)):
            nested_candidates: list[tuple[int, str]] = []
            _collect_answer_candidates(parsed, nested_candidates, path=("json",), depth=0)
            if nested_candidates:
                nested_candidates.sort(key=lambda item: (item[0], len(item[1])), reverse=True)
                return nested_candidates[0][1]

    return ""



def _stream_delta_text(value: Mapping[str, Any]) -> str:
    choices = value.get("choices")
    if not isinstance(choices, Sequence) or isinstance(choices, (str, bytes)):
        return ""

    parts: list[str] = []
    for choice in choices:
        if not isinstance(choice, Mapping):
            continue
        delta = choice.get("delta")
        if isinstance(delta, Mapping) and isinstance(delta.get("content"), str):
            parts.append(delta["content"])
            continue
        message = choice.get("message")
        if isinstance(message, Mapping) and isinstance(message.get("content"), str):
            parts.append(message["content"])
    return "".join(parts)

def _json_loads_safely(value: str) -> Any:
    try:
        return json.loads(value)
    except Exception:
        return None


def _merge_stream_pieces(pieces: list[str]) -> str:
    raw_pieces = [str(piece) for piece in pieces if str(piece or "")]
    cleaned = [_clean_answer(piece, clean_ragflow_markers=True) for piece in raw_pieces if _clean(piece)]
    cleaned = [piece for piece in cleaned if piece]
    if not cleaned:
        return ""
    if len(cleaned) == 1:
        return cleaned[0]

    # RAGFlow-like streams sometimes send cumulative answer states. Prefer the
    # longest final state if every previous piece is contained in a later piece.
    longest = max(cleaned, key=len)
    if all(piece == longest or piece in longest for piece in cleaned):
        return longest

    # OpenAI-style streams usually send deltas. Preserve spaces between deltas.
    if all(len(piece) < 240 for piece in cleaned):
        return _clean_answer("".join(raw_pieces), clean_ragflow_markers=True)

    return cleaned[-1]


def _content_parts_to_text(content: Any) -> str:
    if not isinstance(content, Sequence) or isinstance(content, (str, bytes)):
        return ""

    parts: list[str] = []
    for item in content:
        if isinstance(item, str):
            parts.append(item)
            continue
        if not isinstance(item, Mapping):
            continue
        if isinstance(item.get("text"), str):
            parts.append(item["text"])
            continue
        if isinstance(item.get("content"), str):
            parts.append(item["content"])
            continue
        if isinstance(item.get("value"), str):
            parts.append(item["value"])
    return "\n".join(part for part in parts if part).strip()


def _valid_answer_candidate(text: str) -> bool:
    clean = _clean(text)
    if len(clean) < 2:
        return False
    if clean in {"[]", "{}", "true", "false", "null", "ok", "OK"}:
        return False
    if _looks_like_source_markdown_only(text):
        return False
    if clean.lower().startswith(("## gefundene quellen", "### quellen", "quellenüberblick")):
        return False
    if clean.startswith("[") and clean.endswith("]") and len(clean) < 200:
        return False
    return True

# Dieser Abschnitt verarbeitet verschiedene RAGFlow-/Proxy-Formate und erzeugt
# ein einheitliches Quellenmodell für Citation-Events und Markdown.
def _extract_sources(data: Mapping[str, Any]) -> list[Any]:
    found: list[Any] = []

    def walk(value: Any, depth: int = 0) -> None:
        if depth > 8:
            return

        if isinstance(value, Mapping):
            for key in SOURCE_LIST_KEYS:
                raw = value.get(key)
                if isinstance(raw, list):
                    found.extend(raw)
                elif key in REFERENCE_KEYS and isinstance(raw, Mapping):
                    found.extend(_sources_from_reference(raw))

            for key in REFERENCE_KEYS:
                raw = value.get(key)
                if isinstance(raw, Mapping):
                    found.extend(_sources_from_reference(raw))
                elif isinstance(raw, list):
                    found.extend(raw)

            for child in value.values():
                if isinstance(child, (Mapping, list, tuple)):
                    walk(child, depth + 1)

        elif isinstance(value, Sequence) and not isinstance(value, (str, bytes)):
            for child in value:
                if isinstance(child, (Mapping, list, tuple)):
                    walk(child, depth + 1)

    walk(data)
    return found


def _sources_from_reference(reference: Mapping[str, Any]) -> list[dict[str, Any]]:
    sources: list[dict[str, Any]] = []

    chunks = reference.get("chunks")
    if isinstance(chunks, Mapping):
        for ref_id, chunk in _sorted_reference_items(chunks):
            if isinstance(chunk, Mapping):
                item = dict(chunk)
                item.setdefault("reference_id", str(ref_id))
                item.setdefault("rank", _int_or_none(ref_id))
                item.setdefault("source_metadata", {})
                if isinstance(item["source_metadata"], Mapping):
                    item["source_metadata"] = dict(item["source_metadata"])
                else:
                    item["source_metadata"] = {}
                item["source_metadata"].setdefault("reference_id", str(ref_id))
                sources.append(item)
    elif isinstance(chunks, Sequence) and not isinstance(chunks, (str, bytes)):
        for index, chunk in enumerate(chunks, start=1):
            if isinstance(chunk, Mapping):
                item = dict(chunk)
                item.setdefault("reference_id", str(index))
                item.setdefault("rank", index)
                sources.append(item)

    doc_aggs = reference.get("doc_aggs")
    if not sources and isinstance(doc_aggs, Mapping):
        for index, (_, doc) in enumerate(_sorted_reference_items(doc_aggs), start=1):
            if isinstance(doc, Mapping):
                sources.append(
                    {
                        "rank": index,
                        "name": doc.get("doc_name") or doc.get("document_name") or "Quelle",
                        "document_name": doc.get("doc_name") or doc.get("document_name"),
                        "document_id": doc.get("doc_id") or doc.get("document_id"),
                        "text": f"{doc.get('count')} Treffer im Dokument" if doc.get("count") else "",
                        "source_metadata": dict(doc),
                    }
                )
    elif not sources and isinstance(doc_aggs, Sequence) and not isinstance(doc_aggs, (str, bytes)):
        for index, doc in enumerate(doc_aggs, start=1):
            if isinstance(doc, Mapping):
                sources.append(
                    {
                        "rank": index,
                        "name": doc.get("doc_name") or doc.get("document_name") or "Quelle",
                        "document_name": doc.get("doc_name") or doc.get("document_name"),
                        "document_id": doc.get("doc_id") or doc.get("document_id"),
                        "text": f"{doc.get('count')} Treffer im Dokument" if doc.get("count") else "",
                        "source_metadata": dict(doc),
                    }
                )

    return sources


def _sorted_reference_items(mapping: Mapping[str, Any]) -> list[tuple[Any, Any]]:
    def sort_key(item: tuple[Any, Any]) -> tuple[int, str]:
        key = item[0]
        number = _int_or_none(key)
        return (number if number is not None else 10**9, str(key))

    return sorted(mapping.items(), key=sort_key)


def _normalize_sources(
    sources: Any,
    *,
    show_scores: bool = True,
    show_debug: bool = False,
    limit: int = 20,
    connector_base_url: str = "",
    allow_connector_source_links: bool = False,
) -> list[dict[str, Any]]:
    if not isinstance(sources, Sequence) or isinstance(sources, (str, bytes)):
        return []

    normalized: list[dict[str, Any]] = []
    seen: set[tuple[str, str, str]] = set()
    limit = max(1, min(int(limit or 20), MAX_SOURCE_EVENTS_HARD_LIMIT))

    for rank, raw_source in enumerate(sources, start=1):
        source = _source_mapping(raw_source)
        if not source:
            continue

        hit = _normalize_source(
            source,
            rank,
            connector_base_url=connector_base_url,
            allow_connector_source_links=allow_connector_source_links,
        )
        key = (
            _clean(hit.path or hit.document_id or hit.title),
            _clean(hit.chunk_id or hit.reference_id or ""),
            _clean(hit.snippet)[:220],
        )
        if key in seen:
            continue

        seen.add(key)
        normalized.append(
            _source_event_data(hit, show_scores=show_scores, show_debug=show_debug)
        )

        if len(normalized) >= limit:
            break

    return normalized


def _source_mapping(source: Any) -> dict[str, Any]:
    if isinstance(source, Mapping):
        return dict(source)
    if isinstance(source, str) and source.strip():
        return {"text": source}
    return {}


def _normalize_source(
    source: Mapping[str, Any],
    rank: int,
    *,
    connector_base_url: str = "",
    allow_connector_source_links: bool = False,
) -> SourceHit:
    metadata = _metadata(source)
    document_metadata = source.get("document_metadata")
    if isinstance(document_metadata, Mapping):
        metadata = {**dict(document_metadata), **metadata}

    title = _first_nonempty(
        source.get("name"),
        source.get("title"),
        source.get("document_name"),
        source.get("doc_name"),
        source.get("docnm_kwd"),
        metadata.get("name"),
        metadata.get("title"),
        metadata.get("document_name"),
        metadata.get("doc_name"),
        metadata.get("docnm_kwd"),
        _basename(source.get("path") or metadata.get("path")),
        "Quelle",
    )

    snippet = _clean_snippet(
        _first_nonempty(
            source.get("text"),
            source.get("snippet"),
            source.get("content"),
            source.get("content_with_weight"),
            metadata.get("text"),
            metadata.get("snippet"),
            metadata.get("content"),
            metadata.get("content_with_weight"),
            "",
        )
    )

    score = _first_nonempty(
        source.get("score"),
        source.get("relevance_score"),
        source.get("similarity"),
        source.get("similarity_score"),
        source.get("vector_similarity"),
        metadata.get("score"),
        metadata.get("relevance_score"),
        metadata.get("similarity"),
        metadata.get("similarity_score"),
        metadata.get("vector_similarity"),
        None,
    )

    positions = _first_nonempty(source.get("positions"), metadata.get("positions"), None)
    page = _first_nonempty(
        source.get("page"),
        source.get("page_num"),
        source.get("page_number"),
        metadata.get("page"),
        metadata.get("page_num"),
        metadata.get("page_number"),
        _first_page(positions),
        None,
    )
    line = _first_nonempty(source.get("line"), metadata.get("line"), metadata.get("line_number"), None)
    line_start = _first_nonempty(
        source.get("line_start"),
        source.get("start_line"),
        metadata.get("line_start"),
        metadata.get("start_line"),
        line,
        None,
    )
    line_end = _first_nonempty(
        source.get("line_end"),
        source.get("end_line"),
        metadata.get("line_end"),
        metadata.get("end_line"),
        None,
    )
    section = _first_nonempty(
        source.get("section"),
        source.get("section_title"),
        source.get("heading"),
        metadata.get("section"),
        metadata.get("section_title"),
        metadata.get("heading"),

    )

    chunk_id = _string_or_none(
        _first_nonempty(
            source.get("chunk_id"),
            source.get("id"),
            metadata.get("chunk_id"),
            metadata.get("id"),
            None,
        )
    )
    reference_id = _string_or_none(_first_nonempty(source.get("reference_id"), metadata.get("reference_id"), None))
    document_id = _string_or_none(
        _first_nonempty(
            source.get("document_id"),
            source.get("doc_id"),
            metadata.get("document_id"),
            metadata.get("doc_id"),
            None,
        )
    )

    source_metadata = dict(metadata)
    if positions not in (None, ""):
        source_metadata.setdefault("positions", positions)
    if reference_id:
        source_metadata.setdefault("reference_id", reference_id)
    source_id = _string_or_none(
        _first_nonempty(source.get("source_id"), metadata.get("source_id"), None)
    ) or f"S{_int_or_none(source.get('rank')) or rank}"
    document_name = _string_or_none(
        _first_nonempty(
            source.get("document_name"),
            source.get("doc_name"),
            source.get("docnm_kwd"),
            metadata.get("document_name"),
            metadata.get("doc_name"),
            metadata.get("docnm_kwd"),
            title,
            None,
        )
    )
    path = _string_or_none(_first_nonempty(source.get("path"), metadata.get("path"), None))
    locator_quality = _locator_quality(
        page=page,
        section=section,
        line_start=line_start,
        line_end=line_end,
        positions=positions,
        chunk_id=chunk_id,
        document_id=document_id,
        document_name=document_name,
        path=path,
        snippet=snippet,
    )

    return SourceHit(
        rank=_int_or_none(source.get("rank")) or rank,
        title=str(title),
        snippet=snippet,
        source_id=source_id,
        document_name=document_name,
        dataset_name=_string_or_none(
            _first_nonempty(source.get("dataset_name"), metadata.get("dataset_name"), None)

        page=page,
        line=line,
        line_start=line_start,
        line_end=line_end,
        section=section,
        chunk_id=chunk_id,
        reference_id=reference_id,
        score=score,
        preview_url=_best_source_url(
            source,
            metadata,
            keys=PREVIEW_URL_KEYS,
            connector_base_url=connector_base_url,
            allow_connector=allow_connector_source_links,

        original_url=_best_source_url(
            source,
            metadata,
            keys=ORIGINAL_URL_KEYS,
            connector_base_url=connector_base_url,
            allow_connector=allow_connector_source_links,
        ),
        path=path,
        document_id=document_id,
        dataset_id=_string_or_none(
            _first_nonempty(source.get("dataset_id"), metadata.get("dataset_id"), None)
        ),
        repo_id=_string_or_none(_first_nonempty(source.get("repo_id"), metadata.get("repo_id"), None)),

            _first_nonempty(source.get("file_id"), metadata.get("file_id"), metadata.get("seafile_obj_id"), None)

            _first_nonempty(source.get("mime_type"), source.get("mime"), metadata.get("mime_type"), metadata.get("mime"), None)
        ),
        file_type=_string_or_none(
            _first_nonempty(source.get("file_type"), source.get("doc_type"), metadata.get("file_type"), metadata.get("doc_type"), None)
        ),
        positions=positions,
        locator_quality=locator_quality,
        source_metadata=source_metadata,
    )


def _source_event_data(
    hit: SourceHit,
    *,
    show_scores: bool = True,
    show_debug: bool = False,
) -> dict[str, Any]:
    metadata = _public_metadata(hit, show_scores=show_scores, show_debug=show_debug)
    source_url = hit.preview_url or hit.original_url
    source_name = _source_name_with_location(hit.title, metadata)

    event = {
        "name": hit.title,
        "source_id": hit.source_id,
        "url": source_url,
        "preview_url": hit.preview_url,
        "original_url": hit.original_url,
        "text": hit.snippet,
        "snippet": hit.snippet,
        "document": [hit.snippet or hit.title],
        "rank": hit.rank,
        "source_metadata": metadata,
        "metadata": [metadata],
        "source": {"name": source_name, "url": source_url},
        "locator_quality": hit.locator_quality,
        "relevance_label": _relevance(hit.score),
    }

    if show_scores:
        event["score"] = hit.score
        event["relevance"] = _relevance_text(hit.score)

    return _drop_empty(event)


def _citation_event(source: Mapping[str, Any]) -> dict[str, Any]:
    metadata = _metadata(source)
    source_url = _safe_http_url(
        source.get("preview_url") or source.get("url") or source.get("original_url")
    )
    title = _clean(source.get("name")) or "Quelle"
    document = _clean_snippet(source.get("text") or source.get("snippet") or title)
    source_name = _source_name_with_location(title, metadata)

    citation_metadata = dict(metadata)
    citation_metadata.setdefault("source", title)
    citation_metadata.setdefault("date_accessed", datetime.now(timezone.utc).isoformat())
    if source_url:
        citation_metadata.setdefault("url", source_url)

    return {
        "type": "citation",
        "data": {
            "document": [document or title],
            "metadata": [_drop_empty(citation_metadata)],
            "source": _drop_empty({"name": source_name, "url": source_url}),
        },
    }


def _public_metadata(
    hit: SourceHit,
    *,
    show_scores: bool,
    show_debug: bool,
) -> dict[str, Any]:
    metadata: dict[str, Any] = {}

    for key, value in {
        "rank": hit.rank,
        "source_id": hit.source_id,
        "document_name": hit.document_name,
        "dataset_name": hit.dataset_name,
        "page": hit.page,
        "line": hit.line,
        "line_start": hit.line_start,
        "line_end": hit.line_end,
        "section": hit.section,
        "positions": hit.positions,
        "locator_quality": hit.locator_quality,
        "preview_url": hit.preview_url,
        "original_url": hit.original_url,
        "path": hit.path,
        "file_type": hit.file_type,
        "mime_type": hit.mime_type,
        "file_id": hit.file_id,
        "reference_id": hit.reference_id,
        "relevance_label": _relevance(hit.score),
    }.items():
        if value not in (None, "", [], {}):
            metadata[key] = value

    if show_scores and hit.score not in (None, ""):
        metadata["score"] = hit.score
        metadata["relevance"] = _relevance_text(hit.score)

    if show_debug:
        for key, value in {
            "chunk_id": hit.chunk_id,
            "document_id": hit.document_id,
            "dataset_id": hit.dataset_id,
            "repo_id": hit.repo_id,
        }.items():
            if value not in (None, ""):
                metadata[key] = value

        for key, value in hit.source_metadata.items():
            if key not in metadata and value not in (None, "", [], {}):
                metadata[key] = value

    return metadata

# Antworttext und Quellen werden hier zur sichtbaren Assistant-Nachricht
# zusammengesetzt. UX-Regel: Antwort zuerst, Quellen danach.
def _compose_final_answer(answer: str, sources: list[dict[str, Any]], valves: Pipe.Valves) -> str:
    """Compose the final visible assistant message for OpenWebUI."""
    markdown_mode = _source_markdown_mode(valves.SOURCE_MARKDOWN_MODE)
    if _looks_like_source_markdown_only(answer):
        answer = ""
    else:
        answer = _strip_appended_source_markdown(answer)

    if not answer:
        answer = _retrieval_only_fallback_message(sources, valves)
    elif markdown_mode == "audit":
        answer = _ensure_audit_marker(answer, sources)

    source_block = ""
    if bool(valves.APPEND_SOURCE_OVERVIEW) and not _contains_source_markdown(answer):
        source_block = _source_markdown(
            sources,
            show_scores=valves.SHOW_SOURCE_SCORES,
            show_debug=valves.SHOW_SOURCE_DEBUG,
            mode=markdown_mode,
            show_locator_quality=valves.SHOW_LOCATOR_QUALITY,
        )

    if source_block:
        return f"{answer.rstrip()}\n\n{source_block}".strip()

    return answer.strip()


def _ensure_audit_marker(answer: str, sources: list[dict[str, Any]]) -> str:
    if not answer or not sources or re.search(r"\[S\d+\]", answer):
        return answer
    first = sources[0]
    marker = _source_id(first, _metadata(first))
    return f"{answer.rstrip()} [{marker}]"


def _retrieval_only_fallback_message(sources: list[dict[str, Any]], valves: Pipe.Valves) -> str:
    if not sources:
        return (
            "Ich habe keine belastbare Antwort vom RAGFlow-Proxy erhalten und es wurden "
            "keine passenden Quellen zurückgegeben."
        )

    mode = _clean(valves.RETRIEVAL_ONLY_FALLBACK).lower() or "brief"
    if mode == "sources_only":
        return "Ich habe Quellen gefunden, aber der RAGFlow-Proxy hat keinen KI-Antworttext geliefert."

    if mode == "brief":
        return (
            "Der RAGFlow-Proxy hat passende Dokumentstellen gefunden, aber keinen generierten "
            "Antworttext zurückgegeben. Deshalb kann ich hier noch keine echte Chat-RAG-Antwort anzeigen."
        )

    return (
        "### Antwort nicht generiert\n\n"
        "Der RAGFlow-Proxy hat für diese Anfrage nur Retrieval-Treffer zurückgegeben, "
        "aber keinen KI-Antworttext. Diese Pipe ist auf echte Chat-RAG-Antworten ausgelegt "
        "und sendet bereits `response_mode=chat`, `generate_answer=true`, `retrieval_only=false`, "
        "einen RAG-System-Prompt sowie `extra_body.reference=true`.\n\n"
        "Damit im Chat eine echte Antwort statt einer Suchergebnisliste erscheint, muss der Proxy "
        "mindestens eines dieser Antwortfelder zurückgeben: `answer`, `data.answer` oder "
        "OpenAI-kompatibel `choices[0].message.content`. Wenn weiterhin nur Quellen erscheinen, "
        "läuft der Proxy sehr wahrscheinlich noch im Search-/Retrieval-Modus statt im RAGFlow-Chat-Modus."
    )


def _source_markdown(
    sources: list[dict[str, Any]],
    *,
    show_scores: bool = True,
    show_debug: bool = False,
    mode: str = "compact",
    show_locator_quality: bool = True,
) -> str:
    """Render the optional Markdown source overview appended to the answer."""
    mode = _source_markdown_mode(mode)
    if mode == "none":
        return ""

    if mode == "audit":
        return _source_markdown_audit(
            sources,
            show_scores=show_scores,
            show_debug=show_debug,
            show_locator_quality=show_locator_quality,
        )

    if not sources:
        return ""

    groups = _group_sources(sources)
    if mode == "compact":
        return _source_markdown_compact(groups, show_scores=show_scores)
    return _source_markdown_detailed(groups, show_scores=show_scores, show_debug=show_debug)


def _source_markdown_audit(
    sources: list[dict[str, Any]],
    *,
    show_scores: bool,
    show_debug: bool,
    show_locator_quality: bool,
) -> str:
    lines = ["---", "## Nachweise", ""]
    if not sources:
        lines.extend(
            [
                "Ich habe keine ausreichend belastbare Quelle gefunden, um diese Antwort sicher zu belegen.",
                "",
                "**Nachweisqualität:** keine belastbare Quelle.",
            ]
        )
        return "\n".join(lines).rstrip()

    lines.append(_audit_quality_sentence(sources))
    lines.append("")
    lines.append("| ID | Gestützte Aussage | Dokument | Fundstelle | Relevanz | Öffnen |")
    lines.append("|---|---|---|---|---|---|")

    for source in sources[:MAX_SOURCE_GROUPS_DETAILED]:
        metadata = _metadata(source)
        source_id = _source_id(source, metadata)
        claim = _audit_claim(source, metadata)
        document = _audit_document(source, metadata)
        location = _location(metadata)
        locator_quality = str(metadata.get("locator_quality") or "")
        if show_locator_quality and locator_quality in {"unknown", "snippet_only", "document", "chunk"}:
            location = f"{location} (nicht genauer bestimmbar)"
        relevance = _audit_relevance(source, metadata, show_scores=show_scores)
        actions = _actions(source, metadata) or "kein Direktlink"
        lines.append(
            "| "
            + " | ".join(
                _escape_table_cell(value)
                for value in (source_id, claim, document, location, relevance, actions)
            )
            + " |"
        )

    if len(sources) > MAX_SOURCE_GROUPS_DETAILED:
        remaining = len(sources) - MAX_SOURCE_GROUPS_DETAILED
        lines.append(f"|  | weitere Quellen ausgeblendet | {remaining} zusätzliche Treffer |  |  |  |")

    if _has_conflicting_sources(sources):
        lines.extend(
            [
                "",
                "## Hinweis zur Belastbarkeit",
                "Die Quellenlage enthält als widersprüchlich markierte Hinweise. Prüfe die Nachweise und bevorzugt aktuellere oder fachlich freigegebene Dokumente.",
            ]
        )
    else:
        lines.extend(["", "## Hinweis zur Belastbarkeit", _audit_reliability_note(sources)])

    if show_debug:
        debug_lines = []
        for source in sources[:MAX_SOURCE_GROUPS_DETAILED]:
            metadata = _metadata(source)
            debug = _debug_parts(metadata)
            if debug:
                debug_lines.append(f"- {_source_id(source, metadata)}: {' · '.join(debug)}")
        if debug_lines:
            lines.extend(["", "**Debug:**", *debug_lines])

    return "\n".join(lines).rstrip()


def _source_markdown_compact(groups: list[list[dict[str, Any]]], *, show_scores: bool) -> str:
    lines = ["---", "### Quellenüberblick", ""]
    lines.append("| # | Quelle | Fundstelle | Relevanz |")
    lines.append("|---:|---|---|---|")

    for display_rank, group in enumerate(groups[:MAX_SOURCE_GROUPS_COMPACT], start=1):
        source = group[0]
        metadata = _metadata(source)
        source_cell = _source_table_link(source, metadata)
        location = _escape_table_cell(_location(metadata))
        relevance = _escape_table_cell(_relevance_text(source.get("score") or metadata.get("score")) if show_scores else "—")
        if len(group) > 1:
            location = f"{location}<br>{len(group)} Treffer"
        lines.append(f"| {display_rank} | {source_cell} | {location} | {relevance} |")

    if len(groups) > MAX_SOURCE_GROUPS_COMPACT:
        remaining = len(groups) - MAX_SOURCE_GROUPS_COMPACT
        lines.append(f"|  | weitere Quellen | {remaining} zusätzliche Treffer |  |")

    return "\n".join(lines).rstrip()


def _source_markdown_detailed(
    groups: list[list[dict[str, Any]]],
    *,
    show_scores: bool,
    show_debug: bool,
) -> str:
    lines = ["---", "## Quellen & Fundstellen", ""]

    for display_rank, group in enumerate(groups[:MAX_SOURCE_GROUPS_DETAILED], start=1):
        source = group[0]
        metadata = _metadata(source)
        title = _escape_markdown(source.get("name") or "Quelle")
        summary = _source_summary(metadata, source, len(group), show_scores=show_scores)

        lines.append(f"### {display_rank}. {title}")
        if summary:
            lines.append(f"**{summary}**")

        actions = _actions(source, metadata)
        if actions:
            lines.append(f"**Aktionen:** {actions}")

        snippet = _clean_snippet(source.get("text") or source.get("snippet") or "")
        if snippet:
            lines.append("")
            lines.extend(_blockquote(_compact(snippet, DETAILED_SNIPPET_CHARS)))

        if show_debug:
            debug = _debug_parts(metadata)
            if debug:
                lines.append("")
                lines.append(f"Debug: {' · '.join(debug)}")

        lines.append("")

    return "\n".join(lines).rstrip()


def _source_markdown_mode(value: Any) -> str:
    mode = _clean(value).lower() or "compact"
    if mode in {"off", "disabled"}:
        return "none"
    if mode in VALID_SOURCE_MARKDOWN_MODES:
        return mode
    return "compact"


def _source_summary(
    metadata: Mapping[str, Any],
    source: Mapping[str, Any],
    group_size: int,
    *,
    show_scores: bool,
) -> str:
    parts = [_location(metadata)]

    if show_scores:
        parts.append(_relevance_text(source.get("score") or metadata.get("score")))

    if group_size > 1:
        parts.append(f"{group_size} Treffer")

    return " · ".join(part for part in parts if part)


def _audit_quality_sentence(sources: list[dict[str, Any]]) -> str:
    quality = _audit_quality(sources)
    precise = 0
    linked = 0
    for source in sources:
        metadata = _metadata(source)
        if metadata.get("locator_quality") in {"line", "page", "section", "position"}:
            precise += 1
        if source.get("preview_url") or source.get("url") or source.get("original_url"):
            linked += 1
    labels = {
        "strong": f"stark - {len(sources)} Quellen, {precise} konkrete Fundstellen, {linked} Direktlinks.",
        "medium": f"mittel - {len(sources)} Quellen, {precise} konkrete Fundstellen, {linked} Direktlinks.",
        "weak": f"schwach - {len(sources)} Quellen, aber nur grobe Dokument-/Chunk-Hinweise.",
        "none": "keine belastbare Quelle.",
    }
    return f"**Nachweisqualität:** {labels[quality]}"


def _audit_quality(sources: list[dict[str, Any]]) -> str:
    if not sources:
        return "none"
    precise = 0
    linked = 0
    for source in sources:
        metadata = _metadata(source)
        if metadata.get("locator_quality") in {"line", "page", "section", "position"}:
            precise += 1
        if source.get("preview_url") or source.get("url") or source.get("original_url"):
            linked += 1
    if len(sources) >= 2 and precise >= 2 and linked >= 1:
        return "strong"
    if precise >= 1 or linked >= 1:
        return "medium"
    return "weak"


def _audit_claim(source: Mapping[str, Any], metadata: Mapping[str, Any]) -> str:
    claim = _first_nonempty(
        source.get("claim"),
        source.get("supported_claim"),
        metadata.get("claim"),
        metadata.get("supported_claim"),
        None,
    )
    if claim not in (None, ""):
        return _compact(claim, 120)
    snippet = _clean_snippet(source.get("text") or source.get("snippet") or "")
    if snippet:
        return _compact(snippet, 120)
    return "Gestützte Aussage nicht genauer bestimmbar"


def _audit_document(source: Mapping[str, Any], metadata: Mapping[str, Any]) -> str:
    title = _clean(
        source.get("name")
        or metadata.get("document_name")
        or metadata.get("doc_name")
        or "Quelle"
    )
    path = _clean(metadata.get("path"))
    dataset = _clean(metadata.get("dataset_name"))
    parts = [title]
    if path and path != title:
        parts.append(path)
    if dataset:
        parts.append(dataset)
    return " · ".join(parts)


def _audit_relevance(
    source: Mapping[str, Any],
    metadata: Mapping[str, Any],
    *,
    show_scores: bool,
) -> str:
    if show_scores:
        return _relevance_text(source.get("score") or metadata.get("score"))
    return str(
        metadata.get("relevance_label")
        or _relevance(source.get("score") or metadata.get("score"))
    )


def _audit_reliability_note(sources: list[dict[str, Any]]) -> str:
    quality = _audit_quality(sources)
    if quality == "strong":
        return "Die Antwort basiert auf mehreren Quellen mit konkreten Fundstellen und Direktlinks."
    if quality == "medium":
        return "Die Antwort basiert auf Quellen, deren Fundstellen teilweise konkret bestimmbar sind."
    return "Die Antwort basiert auf Quellen mit grober Fundstelle; prüfe die Preview, bevor du daraus eine Entscheidung ableitest."


def _has_conflicting_sources(sources: list[dict[str, Any]]) -> bool:
    conflict_values = {"conflict", "conflicting", "contradiction", "contradicts", "widerspruch"}
    for source in sources:
        metadata = _metadata(source)
        for mapping in (source, metadata):
            for key, value in mapping.items():
                key_text = str(key).lower()
                value_text = str(value).lower()
                if key_text in conflict_values and value not in (None, "", False):
                    return True
                if key_text in {"assessment", "status", "note"} and "widerspr" in value_text:
                    return True
                if key_text in {"assessment", "status", "note"} and "contradict" in value_text:
                    return True
    return False


def _group_sources(sources: list[dict[str, Any]]) -> list[list[dict[str, Any]]]:
    grouped: dict[str, list[dict[str, Any]]] = {}

    for source in sources:
        metadata = _metadata(source)
        key = _first_nonempty(
            metadata.get("path"),
            metadata.get("document_id"),
            metadata.get("document_name"),
            metadata.get("doc_name"),
            source.get("name"),
            "Quelle",
        )
        grouped.setdefault(str(key), []).append(source)

    groups = list(grouped.values())
    for group in groups:
        group.sort(key=lambda item: (_score_sort(item), _rank_sort(item)))

    groups.sort(key=lambda group: (_score_sort(group[0]), _rank_sort(group[0])))
    return groups


def _score_sort(source: Mapping[str, Any]) -> float:
    score = _score_float(source.get("score") or _metadata(source).get("score"))
    return -1.0 if score is None else -score


def _rank_sort(source: Mapping[str, Any]) -> int:
    try:
        return int(source.get("rank") or _metadata(source).get("rank") or 9999)
    except Exception:
        return 9999


def _metadata(source: Mapping[str, Any]) -> dict[str, Any]:
    metadata = source.get("source_metadata")
    if isinstance(metadata, Mapping):
        return dict(metadata)

    metadata_items = source.get("metadata")
    if (
        isinstance(metadata_items, Sequence)
        and not isinstance(metadata_items, (str, bytes))
        and metadata_items
        and isinstance(metadata_items[0], Mapping)
    ):
        return dict(metadata_items[0])

    if isinstance(metadata_items, Mapping):
        return dict(metadata_items)

    return {}


def _location(metadata: Mapping[str, Any]) -> str:
    parts = []
    if metadata.get("page") not in (None, ""):
        parts.append(f"Seite {metadata.get('page')}")
    if metadata.get("section") not in (None, ""):
        parts.append(f"Abschnitt {metadata.get('section')}")
    line = _line_range(metadata.get("line_start") or metadata.get("line"), metadata.get("line_end"))
    if line:
        parts.append(f"Zeile {line}")
    if not parts and metadata.get("reference_id") not in (None, ""):
        parts.append(f"Referenz {metadata.get('reference_id')}")
    if not parts and metadata.get("positions") not in (None, "", [], {}):
        parts.append("Positionen vorhanden")
    return " · ".join(parts) or "Fundstelle nicht angegeben"


def _actions(source: Mapping[str, Any], metadata: Mapping[str, Any]) -> str:
    links = []

    preview = _safe_http_url(source.get("preview_url") or metadata.get("preview_url"))
    original = _safe_http_url(source.get("original_url") or metadata.get("original_url"))

    if preview:
        links.append(_markdown_link("Preview öffnen", preview))
    if original and original != preview:
        links.append(_markdown_link("Original öffnen", original))

    return " · ".join(link for link in links if link)


def _source_table_link(source: Mapping[str, Any], metadata: Mapping[str, Any]) -> str:
    title = _clean(source.get("name") or metadata.get("document_name") or metadata.get("doc_name") or "Quelle")
    label = _middle_ellipsis(title, 72)
    url = _safe_http_url(source.get("preview_url") or source.get("original_url"))
    if not url:
        return _escape_table_cell(label)
    return f"[{_escape_link_label(label)}](<{url}>)"


def _markdown_link(label: str, url: str) -> str:
    safe_url = _safe_http_url(url)
    if not safe_url:
        return ""
    return f"[{_escape_markdown(label)}](<{safe_url}>)"


def _debug_parts(metadata: Mapping[str, Any]) -> list[str]:
    parts = []
    for label, key in (
        ("Chunk", "chunk_id"),
        ("Dokument", "document_id"),
        ("Dataset", "dataset_id"),
        ("Repo", "repo_id"),
    ):
        value = metadata.get(key)
        if value not in (None, ""):
            parts.append(f"{label} `{_escape_markdown(str(value)[:12])}`")
    return parts


def _relevance_text(score: Any) -> str:
    formatted = _format_score(score)
    if formatted:
        return f"{_relevance(score)} ({formatted})"
    return "unbekannt"


def _relevance(score: Any) -> str:
    value = _score_float(score)
    if value is None:
        return "unbekannt"
    if value >= 0.85:
        return "sehr hoch"
    if value >= 0.7:
        return "hoch"
    if value >= 0.5:
        return "mittel"
    return "niedrig"


def _format_score(score: Any) -> str:
    value = _score_float(score)
    if value is None:
        return ""
    return f"{value:.0%}"


def _score_float(score: Any) -> Optional[float]:
    try:
        value = float(score)
    except Exception:
        return None

    if value > 1:
        value = value / 100

    return max(0.0, min(1.0, value))

# Entfernt technische Marker, HTML-Reste und escapet Markdown, damit Chat und
# Quellenübersicht sauber und professionell dargestellt werden.
def _clean_answer(answer: str, *, clean_ragflow_markers: bool) -> str:
    text = str(answer or "").replace("\r\n", "\n").replace("\r", "\n")
    text = html.unescape(text).replace("\u00a0", " ")

    if clean_ragflow_markers:
        text = re.sub(r"\{\{\s*source\s*:\s*\d+\s*\}\}", "", text, flags=re.IGNORECASE)
        text = re.sub(r"\s*##\d+\$\$", "", text)
        text = re.sub(r"\s*\[\s*ID\s*:\s*\d+\s*\]", "", text, flags=re.IGNORECASE)
        text = re.sub(r"\s*\[\s*source\s*:\s*\d+\s*\]", "", text, flags=re.IGNORECASE)

    text = re.sub(r"[ \t]+\n", "\n", text)
    text = re.sub(r"\n{4,}", "\n\n\n", text)
    return text.strip()


class _SnippetHTMLTextExtractor(HTMLParser):
    def __init__(self) -> None:
        super().__init__(convert_charrefs=True)
        self._parts: list[str] = []
        self._ignored_depth = 0

    @property
    def text(self) -> str:
        return "".join(self._parts)

    def handle_starttag(self, tag: str, _attrs: list[tuple[str, str | None]]) -> None:
        name = tag.lower()
        if name in {"script", "style"}:
            self._ignored_depth += 1
            return
        if self._ignored_depth:
            return
        if name == "br":
            self._parts.append("\n")
        elif name == "tr":
            self._append_line_break()
        elif name in {"td", "th"}:
            self._append_table_cell_separator()

    def handle_endtag(self, tag: str) -> None:
        name = tag.lower()
        if name in {"script", "style"} and self._ignored_depth:
            self._ignored_depth -= 1
            return
        if self._ignored_depth:
            return
        if name == "tr":
            self._append_line_break()

    def handle_data(self, data: str) -> None:
        if not self._ignored_depth:
            self._parts.append(data)

    def _append_line_break(self) -> None:
        if self._parts and self._parts[-1] != "\n":
            self._parts.append("\n")

    def _append_table_cell_separator(self) -> None:
        if not self._parts:
            return
        current = "".join(self._parts).rstrip()
        if current and not current.endswith(("\n", "|")):
            self._parts.append(" | ")


def _clean_snippet(value: Any) -> str:
    clean = str(value or "")
    if not clean:
        return ""

    if "<" in clean or "&" in clean:
        parser = _SnippetHTMLTextExtractor()
        try:
            parser.feed(clean)
            parser.close()
        except Exception:
            return ""
        clean = html.unescape(parser.text).replace("\u00a0", " ")
    clean = clean.replace("\r\n", "\n").replace("\r", "\n")

    lines = (" ".join(line.split()) for line in clean.splitlines())
    return "\n".join(line for line in lines if line).strip()


def _blockquote(text: Any) -> list[str]:
    return [
        f"> {_escape_markdown(line)}" if line else ">"
        for line in str(text or "").splitlines()
    ]


def _error_card(title: str, message: str) -> str:
    return f"### {title}\n\n{message.strip()}"


def _compact(value: Any, limit: int = 220) -> str:
    clean = _clean(value)
    return clean if len(clean) <= limit else clean[: limit - 1].rstrip() + "…"


def _clean(value: Any) -> str:
    return " ".join(str(value or "").split())


def _escape_markdown(value: Any) -> str:
    replacements = {
        "\\": "\\\\",
        "`": "\\`",
        "*": "\\*",
        "_": "\\_",
        "{": "\\{",
        "}": "\\}",
        "[": "\\[",
        "]": "\\]",
        "(": "\\(",
        ")": "\\)",
        "#": "\\#",
        "|": "\\|",
        "<": "&lt;",
        ">": "&gt;",
    }
    return "".join(replacements.get(char, char) for char in str(value or ""))


def _escape_table_cell(value: Any) -> str:
    text = str(value or "")
    replacements = {
        "\\": "\\\\",
        "`": "\\`",
        "|": "\\|",
        "<": "&lt;",
        ">": "&gt;",
    }
    return "".join(replacements.get(char, char) for char in text).replace("\n", "<br>")


def _escape_link_label(value: Any) -> str:
    text = str(value or "")
    replacements = {
        "\\": "\\\\",
        "[": "\\[",
        "]": "\\]",
        "`": "\\`",
        "|": "\\|",
    }
    return "".join(replacements.get(char, char) for char in text)


def _escape_inline_code(value: Any) -> str:
    return str(value or "").replace("`", "ʼ")




def _best_source_url(
    source: Mapping[str, Any],
    metadata: Mapping[str, Any],
    *,
    keys: Sequence[str],
    connector_base_url: str = "",
    allow_connector: bool = False,
) -> Optional[str]:
    """Pick the best real document URL and avoid leaking connector API links."""
    for mapping in (source, metadata):
        if not isinstance(mapping, Mapping):
            continue
        for key in keys:
            url = _allowed_source_url(
                mapping.get(key),
                connector_base_url=connector_base_url,
                allow_connector=allow_connector,
            )
            if url:
                return url
    return None


def _allowed_source_url(value: Any, *, connector_base_url: str = "", allow_connector: bool = False) -> Optional[str]:
    url = _safe_http_url(value)
    if not url:
        return None
    if not allow_connector and _is_connector_url(url, connector_base_url):
        return None
    return url


def _is_connector_url(url: str, connector_base_url: str = "") -> bool:
    parsed = urlsplit(str(url or ""))
    path = (parsed.path or "").lower()

    if path.startswith("/api/openwebui/sources/preview"):
        return False

    if "/api/openwebui/proxy" in path or path.endswith(PROXY_CHAT_PATH):
        return True

    base = urlsplit(_clean(connector_base_url))
    if base.netloc and parsed.netloc.lower() == base.netloc.lower():
        return True

    return False


def _middle_ellipsis(value: Any, limit: int = 72) -> str:
    text = _clean(value)
    if len(text) <= limit:
        return text
    keep = max(8, (limit - 1) // 2)
    tail = max(8, limit - keep - 1)
    return f"{text[:keep].rstrip()}…{text[-tail:].lstrip()}"

def _safe_http_url(value: Any) -> Optional[str]:
    url = str(value or "").strip()
    if not url:
        return None

    parsed = urlsplit(url)
    if parsed.scheme not in {"http", "https"} or not parsed.netloc:
        return None

    return url.replace("\r", "").replace("\n", "").replace(">", "%3E")


def _drop_empty(value: Any) -> Any:
    if isinstance(value, dict):
        return {
            key: _drop_empty(item)
            for key, item in value.items()
            if item not in (None, "", [], {})
        }
    if isinstance(value, list):
        return [_drop_empty(item) for item in value if item not in (None, "", [], {})]
    return value


def _first_nonempty(*values: Any) -> Any:
    for value in values:
        if value not in (None, "", [], {}):
            return value
    return None


def _source_id(source: Mapping[str, Any], metadata: Mapping[str, Any]) -> str:
    value = source.get("source_id") or metadata.get("source_id")
    if value not in (None, ""):
        return str(value)
    rank = source.get("rank") or metadata.get("rank") or 1
    try:
        return f"S{int(rank)}"
    except Exception:
        return "S?"


def _source_name_with_location(title: Any, metadata: Mapping[str, Any]) -> str:
    clean_title = _clean(title) or "Quelle"
    location = _location(metadata)
    if location and location != "Fundstelle nicht angegeben":
        return f"{clean_title} · {location}"
    return clean_title


def _line_range(start: Any, end: Any) -> str:
    if start in (None, "") and end in (None, ""):
        return ""
    if end in (None, "") or str(end) == str(start):
        return str(start)
    if start in (None, ""):
        return str(end)
    return f"{start}-{end}"


def _first_page(positions: Any) -> Any:
    if not isinstance(positions, Sequence) or isinstance(positions, (str, bytes)) or not positions:
        return None
    first = positions[0]
    if isinstance(first, Sequence) and not isinstance(first, (str, bytes)) and first:
        return first[0]
    return None


def _locator_quality(
    *,
    page: Any,
    section: Any,
    line_start: Any,
    line_end: Any,
    positions: Any,
    chunk_id: Optional[str],
    document_id: Optional[str],
    document_name: Optional[str],
    path: Optional[str],
    snippet: str,
) -> str:
    if line_start not in (None, "") or line_end not in (None, ""):
        return "line"
    if page not in (None, ""):
        return "page"
    if section not in (None, ""):
        return "section"
    if positions not in (None, "", [], {}):
        return "position"
    if chunk_id:
        return "chunk"
    if document_id or document_name or path:
        return "document"
    if snippet:
        return "snippet_only"
    return "unknown"


def _string_or_none(value: Any) -> Optional[str]:
    if value in (None, ""):
        return None
    return str(value)


def _int_or_none(value: Any) -> Optional[int]:
    try:
        return int(str(value))
    except Exception:
        return None


def _basename(path: Any) -> Optional[str]:
    clean = _clean(path)
    if not clean:
        return None
    return clean.rstrip("/").rsplit("/", 1)[-1] or None
