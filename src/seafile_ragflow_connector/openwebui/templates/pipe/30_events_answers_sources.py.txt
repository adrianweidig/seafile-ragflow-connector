_EXACT_QUERY_TOKEN_RE = re.compile(
    r"\b(?:"
    r"[A-Z][A-Z0-9]+(?:_[A-Z0-9]+){2,}"
    r"|[A-Fa-f0-9]{12,}"
    r"|[A-Za-z0-9_.-]+\.(?:md|txt|pdf|docx?|xlsx?|csv|json|ya?ml|py|sh|ps1)"
    r"|[A-Z]{2,}-\d{2,}"
    r")\b"
)
_MANAGED_UPLOAD_DOCUMENT_RE = re.compile(
    r"^(?P<stem>.+)\.__connector_"
    r"(?:[0-9a-f]{32}|[0-9a-f]{8}(?:-[0-9a-f]{4}){3}-[0-9a-f]{12})"
    r"(?P<suffix>(?:\.[^/\\]+)?)$",
    re.IGNORECASE,
)


# Event-Emission ist optional und fehlertolerant. Falls ein Frontend ein Event
# nicht akzeptiert, darf die eigentliche Antwort nicht scheitern.
async def _emit_status(
    valves: Pipe.Valves,
    emitter,
    description: str,
    *,
    done: bool,
    hidden: bool = False,
) -> None:
    if not bool(valves.STATUS_UPDATES):
        return

    mode = _clean(getattr(valves, "STATUS_MODE", "minimal")).lower() or "minimal"
    if mode == "off":
        return

    if mode == "minimal" and not done:
        # OpenWebUI kann Status-Historien als sichtbare Bullet-Liste anzeigen.
        # Deshalb zeigen wir im Minimalmodus nur den ersten ruhigen Startstatus
        # und unterdrücken technische Zwischenschritte.
        if "vorbereitet" not in _clean(description).lower():
            return

    await _emit_event(
        emitter,
        {
            "type": "status",
            "data": {
                "description": description,
                "done": done,
                "hidden": bool(hidden),
            },
        },
    )


async def _emit_event(emitter, event: dict[str, Any]) -> bool:
    if not emitter:
        return False

    try:
        await emitter(event)
        return True
    except Exception:
        LOGGER.warning("OpenWebUI event emission failed", exc_info=True)
        return False


def _completion_status(
    source_count: int,
    elapsed_seconds: float,
    *,
    answer_origin: str = "canonical_answer",
    hit_count: Optional[int] = None,
) -> str:
    if source_count == 0:
        source_text = "keine Quellen"
    elif source_count == 1:
        source_text = "1 Quelle"
    else:
        source_text = f"{source_count} Quellen"
    if hit_count is not None and hit_count > source_count:
        source_text = f"{source_text} · {hit_count} Treffer"
    if answer_origin in ANSWER_GENERATED_ORIGINS:
        answer_text = "Antwort generiert"
    elif answer_origin in {"retrieval_only", "source_inventory"}:
        answer_text = "Quellen gefunden, aber keine Antwort generiert"
    else:
        answer_text = "Keine Antwort und keine belastbaren Quellen"
    return f"{answer_text} · {source_text} · {elapsed_seconds:.1f}s"


def _normalize_task(task: Any) -> str:
    if not task:
        return ""

    value = str(task).strip().lower()
    if "." in value:
        value = value.rsplit(".", 1)[-1]
    return value


def _task_response(task: str, task_body: Optional[dict]) -> str:
    text = _last_user_text((task_body or {}).get("messages") or [])

    if "title" in task:
        return _compact_text(text, 80) or "RAGFlow"
    if "tags" in task or "follow_up" in task:
        return "[]"
    if "emoji" in task:
        return "📚"
    if "query" in task or "image_prompt" in task:
        return _compact_text(text, 160)

    return ""


def _last_user_text(messages: Any) -> str:
    if not isinstance(messages, Sequence) or isinstance(messages, (str, bytes)):
        return ""

    for message in reversed(messages):
        if not isinstance(message, Mapping) or message.get("role") != "user":
            continue

        content = message.get("content")
        if isinstance(content, str):
            return content

        if isinstance(content, Sequence) and not isinstance(content, (str, bytes)):
            parts = []
            for item in content:
                if isinstance(item, Mapping):
                    parts.append(str(item.get("text") or ""))
                elif isinstance(item, str):
                    parts.append(item)
            return " ".join(part for part in parts if part)

    return ""


def _compact_text(text: Any, max_length: int) -> str:
    clean = _clean(text)
    if len(clean) <= max_length:
        return clean
    return clean[: max_length - 1].rstrip() + "…"

# Proxy-/RAGFlow-Antworten können viele Formen haben. Dieser Abschnitt sucht
# bewusst nach generiertem Assistant-Text und ignoriert Quellencontainer.
def _extract_answer(data: Mapping[str, Any]) -> str:
    return extract_answer_result(data).answer


def extract_answer_result(data: Mapping[str, Any]) -> AnswerExtractionResult:
    """Extract a generated answer from explicit answer paths only."""
    warnings: list[str] = []
    for path, value, origin in _canonical_answer_candidates(data):
        text = _answer_value_to_text(value)
        if not text:
            continue
        answer = _clean_answer(text, clean_ragflow_markers=False)
        rejection = _answer_rejection_reason(answer)
        if rejection:
            warnings.append(f"{path}: {rejection}")
            continue
        return AnswerExtractionResult(answer=answer, origin=origin, path=path, warnings=warnings)

    origin = "retrieval_only" if _extract_sources(data) else "none"
    return AnswerExtractionResult(answer="", origin=origin, path="", warnings=warnings)


def _canonical_answer_candidates(data: Mapping[str, Any]) -> list[tuple[str, Any, str]]:
    candidates: list[tuple[str, Any, str]] = []

    for key in CANONICAL_ANSWER_KEYS:
        if key in data:
            candidates.append((key, data.get(key), "canonical_answer"))

    nested = data.get("data")
    if isinstance(nested, Mapping) and "answer" in nested:
        candidates.append(("data.answer", nested.get("answer"), "canonical_answer"))

    candidates.extend(_openai_message_candidates(data, prefix=""))
    if isinstance(nested, Mapping):
        candidates.extend(_openai_message_candidates(nested, prefix="data."))

    return candidates


def _openai_message_candidates(
    value: Mapping[str, Any],
    *,
    prefix: str,
) -> list[tuple[str, Any, str]]:
    candidates: list[tuple[str, Any, str]] = []
    choices = value.get("choices")
    if not isinstance(choices, Sequence) or isinstance(choices, (str, bytes)):
        return candidates

    for index, choice in enumerate(choices):
        if not isinstance(choice, Mapping):
            continue
        message = choice.get("message")
        if isinstance(message, Mapping) and "content" in message:
            candidates.append(
                (
                    f"{prefix}choices[{index}].message.content",
                    message.get("content"),
                    "openai_message",
                )
            )
    return candidates


def _answer_value_to_text(value: Any) -> str:
    if isinstance(value, str):
        from_blob = _extract_answer_from_text_blob(value)
        return from_blob or value
    if isinstance(value, Sequence) and not isinstance(value, (str, bytes)):
        return _content_parts_to_text(value)
    return ""


def _answer_rejection_reason(text: str) -> str:
    clean = _clean(text)
    lower = clean.lower()
    if len(clean) < 2:
        return "empty"
    if _looks_like_backend_error_answer(clean):
        return "backend error"
    if clean in {"[]", "{}", "true", "false", "null", "ok", "OK"}:
        return "non-answer scalar"
    if _looks_like_source_markdown_only(text):
        return "source markdown"
    if re.fullmatch(r"\d+\s+treffer\s+im\s+dokument", lower):
        return "document hit count"
    if _looks_like_filename_or_path(clean):
        return "filename or path"
    if _looks_like_markdown_table_without_answer(text):
        return "markdown table without answer"
    if _looks_like_short_title(clean):
        return "short title without sentence structure"
    return ""


def _looks_like_backend_error_answer(text: str) -> bool:
    clean = _clean(text)
    return bool(
        re.fullmatch(
            r"\*{0,3}(?:error|fehler|exception)\*{0,3}\s*:\s*.{0,220}"
            r"\b(?:not\s+authorized|unauthorized|forbidden|permission\s+denied|access\s+denied|api[- ]?key|model\(@none\))\b.*",
            clean,
            flags=re.IGNORECASE,
        )
    )


def _looks_like_filename_or_path(text: str) -> bool:
    clean = _clean(text)
    if not clean or "\n" in str(text):
        return False
    if re.fullmatch(r"[\w .@()+\-/\\]+?\.(md|txt|pdf|docx?|xlsx?|pptx?|csv|json|yaml|yml)", clean, flags=re.IGNORECASE):
        return True
    if re.search(r"(^|[A-Za-z]:|/|\\)[\w .@()+\-/\\]+\.(md|txt|pdf|docx?|xlsx?|pptx?|csv|json|yaml|yml)$", clean, flags=re.IGNORECASE):
        return True
    return False


def _looks_like_markdown_table_without_answer(text: str) -> bool:
    lines = [line.strip() for line in str(text or "").splitlines() if line.strip()]
    if not lines:
        return False
    table_lines = [line for line in lines if line.startswith("|") and line.endswith("|")]
    return len(table_lines) >= 2 and len(table_lines) / len(lines) >= 0.8


def _looks_like_short_title(text: str) -> bool:
    clean = _clean(text)
    if len(clean) > 64:
        return False
    if re.search(r"[.!?;:]", clean):
        return False
    words = clean.split()
    if len(words) <= 1:
        return True
    if len(words) <= 5 and not re.search(
        r"\b(ist|sind|hat|haben|wird|werden|kann|können|muss|müssen|zeigt|beschreibt|regelt|enthält)\b",
        clean,
        flags=re.IGNORECASE,
    ):
        return True
    return False


def _extract_answer_from_text_blob(text: str) -> str:
    raw = str(text or "").strip()
    if not raw:
        return ""

    # Some proxies forward SSE as one string: data: {...}\n\ndata: [DONE]
    if raw.startswith("data:") or "\ndata:" in raw:
        pieces: list[str] = []
        for line in raw.splitlines():
            line = line.strip()
            if not line.startswith("data:"):
                continue
            payload = line[5:].strip()
            if not payload or payload == "[DONE]":
                continue
            parsed = _json_loads_safely(payload)
            if parsed is None:
                continue
            if isinstance(parsed, Mapping):
                piece = _stream_delta_text(parsed) or extract_answer_result(parsed).answer
            elif isinstance(parsed, str):
                piece = parsed
            else:
                piece = ""
            if piece:
                pieces.append(piece)
        return _merge_stream_pieces(pieces)

    # Some proxies double-encode a JSON response as a string.
    if raw[:1] in {"{", "["}:
        parsed = _json_loads_safely(raw)
        if isinstance(parsed, Mapping):
            return extract_answer_result(parsed).answer
        if isinstance(parsed, Sequence) and not isinstance(parsed, (str, bytes)):
            pieces = [
                extract_answer_result(item).answer
                for item in parsed
                if isinstance(item, Mapping)
            ]
            return _merge_stream_pieces(pieces)

    return ""



def _stream_delta_text(value: Mapping[str, Any]) -> str:
    choices = value.get("choices")
    if not isinstance(choices, Sequence) or isinstance(choices, (str, bytes)):
        return ""

    parts: list[str] = []
    for choice in choices:
        if not isinstance(choice, Mapping):
            continue
        delta = choice.get("delta")
        if isinstance(delta, Mapping) and isinstance(delta.get("content"), str):
            parts.append(delta["content"])
            continue
        message = choice.get("message")
        if isinstance(message, Mapping) and isinstance(message.get("content"), str):
            parts.append(message["content"])
    return "".join(parts)

def _json_loads_safely(value: str) -> Any:
    try:
        return json.loads(value)
    except Exception:
        return None


def _merge_stream_pieces(pieces: list[str]) -> str:
    raw_pieces = [str(piece) for piece in pieces if str(piece or "")]
    cleaned = [_clean_answer(piece, clean_ragflow_markers=False) for piece in raw_pieces if _clean(piece)]
    cleaned = [piece for piece in cleaned if piece]
    if not cleaned:
        return ""
    if len(cleaned) == 1:
        return cleaned[0]

    # RAGFlow-like streams sometimes send cumulative answer states. Prefer the
    # longest final state if every previous piece is contained in a later piece.
    longest = max(cleaned, key=len)
    if all(piece == longest or piece in longest for piece in cleaned):
        return longest

    # OpenAI-style streams usually send deltas. Preserve spaces between deltas.
    if all(len(piece) < 240 for piece in cleaned):
        return _clean_answer("".join(raw_pieces), clean_ragflow_markers=False)

    return cleaned[-1]


def _content_parts_to_text(content: Any) -> str:
    if not isinstance(content, Sequence) or isinstance(content, (str, bytes)):
        return ""

    parts: list[str] = []
    for item in content:
        if isinstance(item, str):
            parts.append(item)
            continue
        if not isinstance(item, Mapping):
            continue
        if isinstance(item.get("text"), str):
            parts.append(item["text"])
            continue
        if isinstance(item.get("content"), str):
            parts.append(item["content"])
            continue
        if isinstance(item.get("value"), str):
            parts.append(item["value"])
    return "\n".join(part for part in parts if part).strip()

# Dieser Abschnitt verarbeitet verschiedene RAGFlow-/Proxy-Formate und erzeugt
# ein einheitliches Quellenmodell für Citation-Events und Markdown.
def _extract_sources(data: Mapping[str, Any]) -> list[Any]:
    found: list[Any] = []
    seen_containers: set[int] = set()

    for reference in _choice_reference_values(data):
        _extend_sources_once(found, reference, seen_containers, reference_container=True)

    nested = data.get("data")
    if isinstance(nested, Mapping):
        for key in REFERENCE_KEYS:
            if key in nested:
                _extend_sources_once(
                    found,
                    nested.get(key),
                    seen_containers,
                    reference_container=True,
                )

    for key in REFERENCE_KEYS:
        if key in data:
            _extend_sources_once(found, data.get(key), seen_containers, reference_container=True)

    for container in (data, nested if isinstance(nested, Mapping) else None):
        if not isinstance(container, Mapping):
            continue
        for key in SOURCE_LIST_KEYS:
            if key in container:
                _extend_sources_once(
                    found,
                    container.get(key),
                    seen_containers,
                    reference_container=False,
                )

    return found


def _choice_reference_values(data: Mapping[str, Any]) -> list[Any]:
    references: list[Any] = []
    for container in (data, data.get("data") if isinstance(data.get("data"), Mapping) else None):
        if not isinstance(container, Mapping):
            continue
        choices = container.get("choices")
        if not isinstance(choices, Sequence) or isinstance(choices, (str, bytes)):
            continue
        for choice in choices:
            if not isinstance(choice, Mapping):
                continue
            message = choice.get("message")
            if isinstance(message, Mapping) and "reference" in message:
                references.append(message.get("reference"))
            delta = choice.get("delta")
            if isinstance(delta, Mapping) and "reference" in delta:
                references.append(delta.get("reference"))
    return references


def _extend_sources_once(
    found: list[Any],
    raw: Any,
    seen_containers: set[int],
    *,
    reference_container: bool,
) -> None:
    if raw in (None, ""):
        return
    if isinstance(raw, (Mapping, list, tuple)):
        container_id = id(raw)
        if container_id in seen_containers:
            return
        seen_containers.add(container_id)

    if isinstance(raw, Mapping):
        if reference_container or "chunks" in raw or "doc_aggs" in raw:
            found.extend(_sources_from_reference(raw))
        else:
            found.append(dict(raw))
        return

    if isinstance(raw, Sequence) and not isinstance(raw, (str, bytes)):
        found.extend(raw)


def _sources_from_reference(reference: Mapping[str, Any]) -> list[dict[str, Any]]:
    sources: list[dict[str, Any]] = []

    chunks = reference.get("chunks")
    if isinstance(chunks, Mapping):
        for ref_id, chunk in _sorted_reference_items(chunks):
            if isinstance(chunk, Mapping):
                item = dict(chunk)
                item.setdefault("reference_id", str(ref_id))
                item.setdefault("rank", _int_or_none(ref_id))
                item.setdefault("source_metadata", {})
                if isinstance(item["source_metadata"], Mapping):
                    item["source_metadata"] = dict(item["source_metadata"])
                else:
                    item["source_metadata"] = {}
                item["source_metadata"].setdefault("reference_id", str(ref_id))
                sources.append(item)
    elif isinstance(chunks, Sequence) and not isinstance(chunks, (str, bytes)):
        for index, chunk in enumerate(chunks, start=1):
            if isinstance(chunk, Mapping):
                item = dict(chunk)
                item.setdefault("reference_id", str(index))
                item.setdefault("rank", index)
                sources.append(item)

    doc_aggs = reference.get("doc_aggs")
    if not sources and isinstance(doc_aggs, Mapping):
        for index, (_, doc) in enumerate(_sorted_reference_items(doc_aggs), start=1):
            if isinstance(doc, Mapping):
                sources.append(
                    {
                        "rank": index,
                        "name": doc.get("doc_name") or doc.get("document_name") or "Quelle",
                        "document_name": doc.get("doc_name") or doc.get("document_name"),
                        "document_id": doc.get("doc_id") or doc.get("document_id"),
                        "text": f"{doc.get('count')} Treffer im Dokument" if doc.get("count") else "",
                        "source_metadata": dict(doc),
                    }
                )
    elif not sources and isinstance(doc_aggs, Sequence) and not isinstance(doc_aggs, (str, bytes)):
        for index, doc in enumerate(doc_aggs, start=1):
            if isinstance(doc, Mapping):
                sources.append(
                    {
                        "rank": index,
                        "name": doc.get("doc_name") or doc.get("document_name") or "Quelle",
                        "document_name": doc.get("doc_name") or doc.get("document_name"),
                        "document_id": doc.get("doc_id") or doc.get("document_id"),
                        "text": f"{doc.get('count')} Treffer im Dokument" if doc.get("count") else "",
                        "source_metadata": dict(doc),
                    }
                )

    return sources


def _sorted_reference_items(mapping: Mapping[str, Any]) -> list[tuple[Any, Any]]:
    def sort_key(item: tuple[Any, Any]) -> tuple[int, str]:
        key = item[0]
        number = _int_or_none(key)
        return (number if number is not None else 10**9, str(key))

    return sorted(mapping.items(), key=sort_key)


def _normalize_sources(
    sources: Any,
    *,
    show_scores: bool = True,
    show_debug: bool = False,
    limit: int = 20,
    connector_base_url: str = "",
    allow_connector_source_links: bool = False,
    question: Any = "",
    answer: Any = "",
) -> list[dict[str, Any]]:
    if not isinstance(sources, Sequence) or isinstance(sources, (str, bytes)):
        return []

    hits: list[SourceHit] = []
    seen: set[tuple[str, str, str]] = set()
    limit = max(1, min(int(limit or 20), MAX_SOURCE_EVENTS_HARD_LIMIT))

    for rank, raw_source in enumerate(sources, start=1):
        source = _source_mapping(raw_source)
        if not source:
            continue

        hit = _normalize_source(
            source,
            rank,
            connector_base_url=connector_base_url,
            allow_connector_source_links=allow_connector_source_links,
        )
        key = _source_dedup_key(hit)
        if key in seen:
            continue

        seen.add(key)
        hits.append(hit)

    ranked = _rank_hits_for_audit(hits, question=question, answer=answer)
    return [
        _source_event_data(hit, show_scores=show_scores, show_debug=show_debug)
        for hit in ranked[:limit]
    ]


def _source_dedup_key(hit: SourceHit) -> tuple[str, str, str]:
    document_key = _clean(hit.document_id or hit.path or hit.document_name or hit.title).lower()
    reference_key = _clean(hit.chunk_id or hit.reference_id or "").lower()
    snippet_key = sha256(_clean(hit.snippet).lower().encode("utf-8")).hexdigest()[:20] if hit.snippet else ""
    position_key = _clean(hit.page or hit.line_start or hit.line or hit.positions or "").lower()
    return (document_key, reference_key, f"{snippet_key}:{position_key}")


def _rank_hits_for_audit(
    hits: list[SourceHit],
    *,
    question: Any = "",
    answer: Any = "",
) -> list[SourceHit]:
    exact_terms = _exact_query_terms(question)
    cited_provider_ids = _answer_provider_citation_ids(answer)
    cited_source_labels = _answer_source_labels(answer)
    preserve_source_labels = bool(cited_source_labels) and not cited_provider_ids
    evaluated: list[SourceHit] = []

    for hit in hits:
        components = _audit_score_components(hit, exact_terms=exact_terms)
        provider_used = (
            hit.provider_citation_id is not None
            and hit.provider_citation_id in cited_provider_ids
        )
        label_used = bool(hit.source_id and hit.source_id.upper() in cited_source_labels)
        used_in_answer = provider_used or label_used
        audit_score = _combined_audit_score(components, used_in_answer=used_in_answer)
        match_type = _match_type_from_components(components)
        role = _preliminary_source_role(
            components,
            audit_score=audit_score,
            used_in_answer=used_in_answer,
        )
        evaluated.append(
            replace(
                hit,
                source_role=role,
                match_type=match_type,
                audit_score=audit_score,
                score_components=components,
                used_in_answer=used_in_answer,
                claim_ids=("C1",) if used_in_answer or components["exact_match"] >= 1.0 else (),
                support_status="supported" if used_in_answer or components["exact_match"] >= 1.0 else "related",
            )
        )

    if preserve_source_labels:
        evaluated.sort(key=lambda hit: hit.rank)
    else:
        evaluated.sort(key=_audit_hit_sort_key)
    primary_assigned = False
    final_hits: list[SourceHit] = []
    for display_rank, hit in enumerate(evaluated, start=1):
        role = hit.source_role
        if role == "primary":
            if primary_assigned:
                role = "supporting"
            else:
                primary_assigned = True
        rank = hit.rank if preserve_source_labels else display_rank
        source_id = hit.source_id if preserve_source_labels and hit.source_id else f"S{display_rank}"
        final_hits.append(
            replace(
                hit,
                rank=rank,
                source_id=source_id,
                source_role=role,
            )
        )
    return final_hits


def _exact_query_terms(question: Any) -> tuple[str, ...]:
    text = str(question or "")
    terms: list[str] = []
    terms.extend(match.group(0) for match in _EXACT_QUERY_TOKEN_RE.finditer(text))
    for pattern in (r"`([^`]{4,160})`", r'"([^"]{4,160})"'):
        terms.extend(match.group(1) for match in re.finditer(pattern, text))
    normalized: list[str] = []
    seen: set[str] = set()
    for term in terms:
        clean_term = _clean(term)
        if len(clean_term) < 4:
            continue
        key = clean_term.casefold()
        if key in seen:
            continue
        seen.add(key)
        normalized.append(clean_term)
    return tuple(normalized[:12])


def _answer_provider_citation_ids(answer: Any) -> set[int]:
    text = str(answer or "")
    ids = {
        int(match.group(1))
        for match in re.finditer(r"\[\s*ID\s*:\s*(\d+)\s*\]", text, flags=re.IGNORECASE)
    }
    ids.update(
        int(match.group(1))
        for match in re.finditer(r"\{\{\s*source\s*:\s*(\d+)\s*\}\}", text, flags=re.IGNORECASE)
    )
    ids.update(int(match.group(1)) for match in re.finditer(r"##(\d+)\$\$", text))
    return ids


def _answer_source_labels(answer: Any) -> set[str]:
    return {
        f"S{match.group(1)}".upper()
        for match in re.finditer(r"\[S(\d+)\]", str(answer or ""), flags=re.IGNORECASE)
    }


def _curate_sources_for_answer(sources: list[dict[str, Any]], answer: Any) -> list[dict[str, Any]]:
    if not sources or not answer:
        return sources
    cited_labels = _answer_source_labels(answer)
    cited_provider_ids = _answer_provider_citation_ids(answer)
    if not cited_labels and not cited_provider_ids:
        return sources

    cited_document_keys: set[str] = set()
    for source in sources:
        metadata = _metadata(source)
        source_label = _source_label_for_curation(source, metadata)
        provider_id = _int_or_none(metadata.get("provider_citation_id") or metadata.get("reference_id"))
        if source_label in cited_labels or (
            provider_id is not None and provider_id in cited_provider_ids
        ):
            document_key = _source_document_key_for_curation(source, metadata)
            if document_key:
                cited_document_keys.add(document_key)

    if not cited_document_keys:
        return sources
    curated = [
        source
        for source in sources
        if _source_document_key_for_curation(source, _metadata(source)) in cited_document_keys
    ]
    return curated or sources


def _source_label_for_curation(source: Mapping[str, Any], metadata: Mapping[str, Any]) -> str:
    return str(
        source.get("source_id")
        or source.get("citation_label")
        or metadata.get("source_id")
        or metadata.get("citation_label")
        or ""
    ).upper()


def _source_document_key_for_curation(source: Mapping[str, Any], metadata: Mapping[str, Any]) -> str:
    for value in (
        metadata.get("document_id"),
        metadata.get("path"),
        metadata.get("document_name"),
        metadata.get("doc_name"),
        source.get("name"),
    ):
        text = _clean(value).casefold()
        if text:
            return text
    return ""


def _audit_score_components(hit: SourceHit, *, exact_terms: tuple[str, ...]) -> dict[str, float]:
    source_text = _source_match_text(hit).casefold()
    exact_match = 0.0
    keyword = 0.0
    if exact_terms:
        matches = sum(1 for term in exact_terms if term.casefold() in source_text)
        keyword = matches / len(exact_terms)
        exact_match = 1.0 if matches else 0.0
    vector = _score_float(hit.score) or 0.0
    raw = hit.source_metadata
    reranker = _score_float(
        raw.get("reranker_score")
        or raw.get("rerank_score")
        or raw.get("relevance_score")
        or raw.get("score")
        or raw.get("similarity")
    )
    if reranker is None:
        reranker = vector
    locator = _locator_quality_score(hit.locator_quality)
    authority = 1.0 if hit.original_url or hit.path or hit.document_id else 0.5
    return {
        "exact_match": round(exact_match, 3),
        "keyword": round(keyword, 3),
        "vector": round(vector, 3),
        "reranker": round(reranker, 3),
        "locator_quality": round(locator, 3),
        "source_authority": round(authority, 3),
        "duplicate_penalty": 0.0,
        "weak_locator_penalty": 0.1 if hit.locator_quality in {"unknown", "snippet_only"} else 0.0,
    }


def _combined_audit_score(components: dict[str, float], *, used_in_answer: bool) -> float:
    score = (
        0.35 * components["exact_match"]
        + 0.25 * components["reranker"]
        + 0.15 * components["keyword"]
        + 0.15 * components["vector"]
        + 0.05 * components["locator_quality"]
        + 0.05 * components["source_authority"]
        - components["duplicate_penalty"]
        - components["weak_locator_penalty"]
    )
    if used_in_answer:
        score += 0.15
    if components["exact_match"] >= 1.0:
        score = max(score, 0.95)
    return round(max(0.0, min(1.0, score)), 3)


def _match_type_from_components(components: dict[str, float]) -> str:
    if components["exact_match"] >= 1.0:
        return "exact_string_match"
    if components["keyword"] > 0:
        return "keyword_match"
    if components["reranker"] >= 0.7 and components["vector"] >= 0.5:
        return "hybrid_reranked"
    if components["vector"] > 0:
        return "semantic"
    return "unknown"


def _preliminary_source_role(
    components: dict[str, float],
    *,
    audit_score: float,
    used_in_answer: bool,
) -> str:
    if used_in_answer or components["exact_match"] >= 1.0:
        return "primary"
    if audit_score >= 0.7:
        return "supporting"
    if audit_score < 0.25:
        return "unused"
    return "related"


def _audit_hit_sort_key(hit: SourceHit) -> tuple[int, float, int]:
    role_order = {
        "primary": 0,
        "supporting": 1,
        "related": 2,
        "unused": 3,
        "duplicate": 4,
        "rejected": 5,
    }
    return (role_order.get(hit.source_role, 6), -(hit.audit_score or 0.0), hit.rank)


def _source_match_text(hit: SourceHit) -> str:
    parts = [
        hit.title,
        hit.document_name,
        hit.path,
        hit.snippet,
        hit.chunk_id,
        hit.document_id,
    ]
    for key in ("document_name", "doc_name", "path", "source_path", "title", "name"):
        value = hit.source_metadata.get(key)
        if value not in (None, ""):
            parts.append(str(value))
    return "\n".join(str(part) for part in parts if part not in (None, ""))


def _locator_quality_score(value: Any) -> float:
    return {
        "line": 1.0,
        "page": 0.9,
        "section": 0.85,
        "position": 0.8,
        "chunk": 0.65,
        "document": 0.45,
        "snippet_only": 0.35,
        "unknown": 0.2,
    }.get(str(value or "unknown"), 0.2)


def _source_mapping(source: Any) -> dict[str, Any]:
    if isinstance(source, Mapping):
        return dict(source)
    if isinstance(source, str) and source.strip():
        return {"text": source}
    return {}


def _user_facing_document_name(value: Any, path: Any = None) -> str:
    clean_path = str(path or "").replace("\\", "/").rstrip("/")
    clean = clean_path.rsplit("/", 1)[-1].strip() if clean_path else _clean(value)
    match = _MANAGED_UPLOAD_DOCUMENT_RE.fullmatch(clean)
    if match:
        clean = f"{match.group('stem')}{match.group('suffix')}"
    return clean or "Quelle"


def _normalize_source(
    source: Mapping[str, Any],
    rank: int,
    *,
    connector_base_url: str = "",
    allow_connector_source_links: bool = False,
) -> SourceHit:
    metadata = _metadata(source)
    document_metadata = source.get("document_metadata")
    if isinstance(document_metadata, Mapping):
        metadata = {**dict(document_metadata), **metadata}

    title = _first_nonempty(
        source.get("name"),
        source.get("title"),
        source.get("document_name"),
        source.get("doc_name"),
        source.get("docnm_kwd"),
        metadata.get("name"),
        metadata.get("title"),
        metadata.get("document_name"),
        metadata.get("doc_name"),
        metadata.get("docnm_kwd"),
        _basename(source.get("path") or metadata.get("path")),
        "Quelle",
    )

    snippet = _clean_snippet(
        _first_nonempty(
            source.get("text"),
            source.get("snippet"),
            source.get("content"),
            source.get("content_with_weight"),
            metadata.get("text"),
            metadata.get("snippet"),
            metadata.get("content"),
            metadata.get("content_with_weight"),
            "",
        )
    )

    score = _first_nonempty(
        source.get("score"),
        source.get("relevance_score"),
        source.get("similarity"),
        source.get("similarity_score"),
        source.get("vector_similarity"),
        metadata.get("score"),
        metadata.get("relevance_score"),
        metadata.get("similarity"),
        metadata.get("similarity_score"),
        metadata.get("vector_similarity"),
        None,
    )

    positions = _first_nonempty(source.get("positions"), metadata.get("positions"), None)
    page = _first_nonempty(
        source.get("page"),
        source.get("page_num"),
        source.get("page_number"),
        metadata.get("page"),
        metadata.get("page_num"),
        metadata.get("page_number"),
        _first_page(positions),
        None,
    )
    line = _first_nonempty(source.get("line"), metadata.get("line"), metadata.get("line_number"), None)
    line_start = _first_nonempty(
        source.get("line_start"),
        source.get("start_line"),
        metadata.get("line_start"),
        metadata.get("start_line"),
        line,
        None,
    )
    line_end = _first_nonempty(
        source.get("line_end"),
        source.get("end_line"),
        metadata.get("line_end"),
        metadata.get("end_line"),
        None,
    )
    section = _first_nonempty(
        source.get("section"),
        source.get("section_title"),
        source.get("heading"),
        metadata.get("section"),
        metadata.get("section_title"),
        metadata.get("heading"),
        None,
    )

    chunk_id = _string_or_none(
        _first_nonempty(
            source.get("chunk_id"),
            source.get("id"),
            metadata.get("chunk_id"),
            metadata.get("id"),
            None,
        )
    )
    reference_id = _string_or_none(_first_nonempty(source.get("reference_id"), metadata.get("reference_id"), None))
    provider_citation_id = _int_or_none(
        _first_nonempty(
            source.get("provider_citation_id"),
            source.get("citation_id"),
            metadata.get("provider_citation_id"),
            metadata.get("citation_id"),
            reference_id,
            None,
        )
    )
    document_id = _string_or_none(
        _first_nonempty(
            source.get("document_id"),
            source.get("doc_id"),
            metadata.get("document_id"),
            metadata.get("doc_id"),
            None,
        )
    )

    source_metadata = dict(metadata)
    if positions not in (None, ""):
        source_metadata.setdefault("positions", positions)
    if reference_id:
        source_metadata.setdefault("reference_id", reference_id)
    if provider_citation_id is not None:
        source_metadata.setdefault("provider_citation_id", provider_citation_id)
    source_id = _string_or_none(
        _first_nonempty(source.get("source_id"), metadata.get("source_id"), None)
    ) or f"S{_int_or_none(source.get('rank')) or rank}"
    document_name = _string_or_none(
        _first_nonempty(
            source.get("document_name"),
            source.get("doc_name"),
            source.get("docnm_kwd"),
            metadata.get("document_name"),
            metadata.get("doc_name"),
            metadata.get("docnm_kwd"),
            title,
            None,
        )
    )
    path = _string_or_none(_first_nonempty(source.get("path"), metadata.get("path"), None))
    document_name = _user_facing_document_name(document_name, path)
    title = _user_facing_document_name(title, path)
    locator_quality = _locator_quality(
        page=page,
        section=section,
        line_start=line_start,
        line_end=line_end,
        positions=positions,
        chunk_id=chunk_id,
        document_id=document_id,
        document_name=document_name,
        path=path,
        snippet=snippet,
    )

    return SourceHit(
        rank=_int_or_none(source.get("rank")) or rank,
        title=str(title),
        snippet=snippet,
        source_id=source_id,
        document_name=document_name,
        dataset_name=_string_or_none(
            _first_nonempty(source.get("dataset_name"), metadata.get("dataset_name"), None)
        ),
        page=page,
        line=line,
        line_start=line_start,
        line_end=line_end,
        section=section,
        chunk_id=chunk_id,
        reference_id=reference_id,
        score=score,
        preview_url=_best_source_url(
            source,
            metadata,
            keys=PREVIEW_URL_KEYS,
            connector_base_url=connector_base_url,
            allow_connector=allow_connector_source_links,
        ),
        original_url=_best_source_url(
            source,
            metadata,
            keys=ORIGINAL_URL_KEYS,
            connector_base_url=connector_base_url,
            allow_connector=allow_connector_source_links,
        ),
        path=path,
        document_id=document_id,
        dataset_id=_string_or_none(
            _first_nonempty(source.get("dataset_id"), metadata.get("dataset_id"), None)
        ),
        repo_id=_string_or_none(_first_nonempty(source.get("repo_id"), metadata.get("repo_id"), None)),
        file_id=_string_or_none(
            _first_nonempty(source.get("file_id"), metadata.get("file_id"), metadata.get("seafile_obj_id"), None)
        ),
        mime_type=_string_or_none(
            _first_nonempty(source.get("mime_type"), source.get("mime"), metadata.get("mime_type"), metadata.get("mime"), None)
        ),
        file_type=_string_or_none(
            _first_nonempty(source.get("file_type"), source.get("doc_type"), metadata.get("file_type"), metadata.get("doc_type"), None)
        ),
        positions=positions,
        locator_quality=locator_quality,
        provider_citation_id=provider_citation_id if provider_citation_id is not None else rank - 1,
        source_role=str(source.get("source_role") or metadata.get("source_role") or metadata.get("role") or "related"),
        match_type=str(source.get("match_type") or metadata.get("match_type") or "semantic"),
        audit_score=_score_float(source.get("audit_score") or metadata.get("audit_score")),
        score_components=dict(source.get("score_components") or metadata.get("score_components") or {}),
        used_in_answer=bool(source.get("used_in_answer") or metadata.get("used_in_answer")),
        claim_ids=tuple(source.get("claim_ids") or metadata.get("claim_ids") or ()),
        support_status=str(source.get("support_status") or metadata.get("support_status") or "not_evaluated"),
        source_dto_version=str(
            source.get("source_dto_version")
            or metadata.get("source_dto_version")
            or "v1"
        ),
        status=str(source.get("status") or metadata.get("status") or "available"),
        source_metadata=source_metadata,
    )


def _source_event_data(
    hit: SourceHit,
    *,
    show_scores: bool = True,
    show_debug: bool = False,
) -> dict[str, Any]:
    metadata = _public_metadata(hit, show_scores=show_scores, show_debug=show_debug)
    metadata["source_dto_version"] = hit.source_dto_version
    metadata["status"] = hit.status
    source_url = hit.preview_url or hit.original_url
    source_name = _source_name_with_location(hit.title, metadata)

    event = {
        "source_dto_version": hit.source_dto_version,
        "status": hit.status,
        "name": hit.title,
        "source_id": hit.source_id,
        "url": source_url,
        "preview_url": hit.preview_url,
        "original_url": hit.original_url,
        "text": hit.snippet,
        "snippet": hit.snippet,
        "document": [hit.snippet or hit.title],
        "rank": hit.rank,
        "role": hit.source_role,
        "source_role": hit.source_role,
        "match_type": hit.match_type,
        "audit_score": hit.audit_score,
        "score_components": hit.score_components,
        "used_in_answer": hit.used_in_answer,
        "claim_ids": list(hit.claim_ids),
        "support_status": hit.support_status,
        "source_metadata": metadata,
        "metadata": [metadata],
        "source": {"name": source_name, "url": source_url},
        "locator_quality": hit.locator_quality,
        "relevance_label": _relevance(hit.score),
    }

    if show_scores:
        event["score"] = hit.score
        event["relevance"] = _relevance_text(hit.score)

    return _drop_empty(event)


def _citation_event(source: Mapping[str, Any]) -> dict[str, Any]:
    metadata = _metadata(source)
    source_url = _safe_http_url(
        source.get("preview_url") or source.get("url") or source.get("original_url")
    )
    title = _clean(source.get("name")) or "Quelle"
    document = _clean_snippet(source.get("text") or source.get("snippet") or title)
    source_name = _source_name_with_location(title, metadata)

    citation_metadata = dict(metadata)
    citation_metadata.setdefault("source", title)
    citation_metadata.setdefault("date_accessed", datetime.now(timezone.utc).isoformat())
    if source_url:
        citation_metadata.setdefault("url", source_url)

    return {
        "type": "citation",
        "data": {
            "document": [document or title],
            "metadata": [_drop_empty(citation_metadata)],
            "source": _drop_empty({"name": source_name, "url": source_url}),
        },
    }


def _public_metadata(
    hit: SourceHit,
    *,
    show_scores: bool,
    show_debug: bool,
) -> dict[str, Any]:
    metadata: dict[str, Any] = {}

    for key, value in {
        "rank": hit.rank,
        "source_id": hit.source_id,
        "document_name": hit.document_name,
        "dataset_name": hit.dataset_name,
        "page": hit.page,
        "line": hit.line,
        "line_start": hit.line_start,
        "line_end": hit.line_end,
        "section": hit.section,
        "positions": hit.positions,
        "locator_quality": hit.locator_quality,
        "preview_url": hit.preview_url,
        "original_url": hit.original_url,
        "path": hit.path,
        "file_type": hit.file_type,
        "mime_type": hit.mime_type,
        "file_id": hit.file_id,
        "reference_id": hit.reference_id,
        "provider_citation_id": hit.provider_citation_id,
        "provider_citation_marker": f"[ID:{hit.provider_citation_id}]" if hit.provider_citation_id is not None else None,
        "citation_id": hit.rank - 1,
        "citation_marker": f"[S{hit.rank}]",
        "source_role": hit.source_role,
        "role": hit.source_role,
        "match_type": hit.match_type,
        "audit_score": hit.audit_score,
        "score_components": hit.score_components or None,
        "used_in_answer": hit.used_in_answer,
        "claim_ids": list(hit.claim_ids) if hit.claim_ids else None,
        "support_status": hit.support_status,
        "relevance_label": _relevance(hit.score),
    }.items():
        if value not in (None, "", [], {}):
            metadata[key] = value

    if show_scores and hit.score not in (None, ""):
        metadata["score"] = hit.score
        metadata["relevance"] = _relevance_text(hit.score)

    if show_debug:
        for key, value in {
            "chunk_id": hit.chunk_id,
            "document_id": hit.document_id,
            "dataset_id": hit.dataset_id,
            "repo_id": hit.repo_id,
        }.items():
            if value not in (None, ""):
                metadata[key] = value

        for key, value in hit.source_metadata.items():
            if key not in metadata and value not in (None, "", [], {}):
                metadata[key] = value

    return metadata
